179 void rfc_to_tilt(EST_Relation &ev)
180 {
181  EST_Item *e;
182  EST_Features f;
183 
184  if (ev.f("intonation_style") != "rfc")
185  EST_error("Can't create Tilt parameters from intonation style: %s\n",
186  (const char *)ev.f.S("intonation_style"));
187 
188  for (e = ev.head(); e != 0; e = e->next())

374 float unit_curve(float amp, float dur, float t)
375 {
376  float val;
377  float x;
378 
379  x = (t / (dur)) * 2.0;
380  if (x < 1.0)
381  val = pow(x, float(2.0));
382  else
383  val = 2 - pow((float(2.0) - x), float(2.0));
384 
385  val = (val / 2.0);
386 
387  val *= amp;
388  val += 0; // vert dist.
389 
390  return val;
391 }

393 float fncurve(float length, float t, float curve)
394 {
395  float val;
396  float x;
397 
398  x = (t / length) * 2.0;
399 
400  if (x < 1.0)
401  val = pow(x, curve);
402  else
403  val = 2 - pow((2 - x), curve);
404 
405  val = val / 2.0;
406 
407  return val;
408 }
