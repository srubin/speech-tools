242static void map_to_channels(EST_StrList &channel_map,
243 EST_StrList &channel_names)
244{
245 EST_Litem *p;
246 EST_String b, type, first, last;
247 int n_f, n_l;
248
249 for (p = channel_map.head(); p; p = p->next())
250 {
251 b = channel_map(p);
252 if (b.matches("$", 0))
253 {
254 // do this backwards as types may have "_" in them
255 b = b.after("$");
256 if (!b.contains("-"))
257 {
258 cerr<<"Ill formed coefficient range in map: " << b << "\n";
259 return;
260 }
261 type = b.before("-");
262 first = b.after("-");
263
264 if (!first.contains("+"))
265 {
266 cerr<<"Ill formed coefficient range in map: "<<first<<"\n";
267 return;
268 }
269
270 last = first.after("+");
271 first = first.before("+");
272
273 n_f = Stringtoi(first);
274 n_l = Stringtoi(last);
275
276 for (int i = n_f; i < n_l; ++i)
277 channel_names.append(type + "_" + itoString(i));
278 channel_names.append(type + "_N");
279 }
280 else
281 channel_names.append(b);
282 }
283}

331{
332 int i, j, k;
333
334 if (num_frames() == 0) // i.e. no existing EST_Track to add to
335 {
336 *this = a;
337 return *this;
338 }
339
340 if (a.num_channels() != num_channels())
341 {
342 cerr << "Error: Tried to add " << a.num_channels() <<
343 " channel EST_Track to "<<num_channels() << " channel EST_Track\n";
344 return *this;
345 }
346
347 int old_num = num_frames();
348 float old_end = end();
349 this->resize(a.num_frames()+ this->num_frames(), this->num_channels());
350 for (i = 0, j = old_num; i < a.num_frames(); ++i, ++j)

352 for (k = 0; k < num_channels(); ++k)
353 p_values.a_no_check(j, k) = a(i, k);

439float &EST_Track::a(float t, int c, EST_InterpType interp)
440{
441 static float ia = 0.0;
442
443 if (interp == it_nearest)
444 return p_values.a_no_check(index(t), c);
445 else if (interp == it_linear)
446 {
447 int i = index_below(t);
448 if (i < 0)
449 return a(0,c);
450
451 float n = a(i,c), n1 = a(i+1,c);
452 float tn = p_times(i), tn1 = p_times(i+1);
453 ia = n + (n1-n)*(t-tn)/(tn1-tn);
454 return ia;
455 }
456 else if (interp == it_linear_nz)
457 {
458 int i = index_below(t);
459 if (i < 0)
460 return a(0,c);
461
462 float n = a(i,c), n1 = a(i+1,c);
463 if (fabs(n) < EPSILON(0.0001) || fabs(n1) < EPSILON(0.0001))
464 return p_values.a_no_check(index(t), c);
465 float tn = p_times(i), tn1 = p_times(i+1);
466 ia = n + (n1-n)*(t-tn)/(tn1-tn);
467 return ia;
468 }
469 return ia;
470}

472int EST_Track::index(float x) const
473{
474 if (equal_space())
475 {
476 float s = shift();
477 int f = (int)( (x-t(0))/s+0.5 ); //don't assume track starts at t=0.0
478 if (f<0)
479 f=0;
480 else if (f>= num_frames())
481 f=num_frames()-1;
482 return f;
483 }
484 else if (num_frames() > 1) //if single frame, return that index (0)
485 {
486 int bst, bmid, bend;
487 bst = 1;
488 bend = num_frames();
489 if (x < p_times.a_no_check(bst))
490 bmid=bst;
Value stored to 'bmid' is never read
491 if (x >= p_times.a_no_check(bend-1))
492 bmid=bend-1;
493 else
494 {
495 while (1)
496 {
497 bmid = bst + (bend-bst)/2;
498 if (bst == bmid)
499 break;
500 else if (x < p_times.a_no_check(bmid))
501 {
502 if (x >= p_times.a_no_check(bmid-1))
503 break;
504 bend = bmid;
505 }
506 else
507 bst = bmid;
508 }
509 }
510 if (fabs(x - p_times.a_no_check(bmid)) <
511 fabs(x - p_times.a_no_check(bmid-1)))
512 return bmid;
513 else
514 return bmid - 1;
515 }
516
517 return num_frames() -1;
518}

545/*
546 "p_equal_space" indicates whether the x-axis values are evenly spaced
547 (FIXED) or spaced arbitrarily (VARI).
548
549 "p_single_break" describes the break format. F0 contours are seldom
550 continuous - often breaks occur due to unvoicing etc. These are a
551 marked in the data arrays by break values, "i_break" for ints
552 and "f_break" for floats. The "p_single_break" channel specifies whether
553 a break is represented by a single break value, or as a break
554 value for every frame. eg
555 (SINGLE)
556 800 100
557 810 105
558 BREAK BREAK
559 850 130
560 860 135
561 or
562 (MANY)
563 800 100
564 810 105
565 820 BREAK
566 830 BREAK
567 840 BREAK
568 850 130
569 860 135
570
571 In the MANY case, only the y value is specified as a break, in the
572 SINGLE case the x value may or may not be specified as a break. For
573 this reason, when checking for breaks, it is useful to only rely
574 on the y value being set to the i_break value. Not that if the single_break
575 is MANY and the equal_space is FIXED, you dont really need x-axis
576 values.
577
578 Different functions naturally work better on different representations
579 and that is why all these different types are supported. A
580 general function mod_cont() is supplied to change from one
581 type to another. Not all conversions are currently
582 supported however.
583
584 */

601float EST_Track::shift() const
602{
603 int j1 = 0;
604 int j2 = 0;
605
606 if (!p_equal_space)
607 EST_error(EST_error_where = __null), (*EST_error_func)("Tried to take shift from non-fixed contour\n");
608
609 do
610 {
611 j1 = next_non_break(++j1);
612 j2 = next_non_break(j1);
613 // cout << "j1:" << j1 << " j2:" << j2 << endl;
614 }
615 while ((j2 != 0) && (j2 != (j1 +1)));
616
617 if (j2 == 0)
618 {
619 if (num_frames() > 1)
620 return p_times(1) - p_times(0);
621 else
622 EST_error(EST_error_where = __null), (*EST_error_func)("Couldn't determine shift size\n");
623
624 }
625 return (p_times(j2) - p_times(j1));
626}

673void EST_Track::sample(float f_interval)
674{
675 EST_FVector new_times;
676 EST_FMatrix new_values;
677 EST_CVector new_is_break;
678 int i, j, n;
679
680 n = (int) rint(((end())/ f_interval));
681
682 new_times.resize(n);
683 new_values.resize(n, num_channels());
684 new_is_break.resize(n);
685
686 // REORG - can this be replaced with fill_time()?
687 for (i = 0; i < n; ++i)
688 new_times[i] = (float) ((i + 1) * f_interval);
689
690 for (i = 0; i < n; ++i)
691 {
692 new_is_break[i] = !interp_value(new_times(i), f_interval);
693 for (j = 0; j < num_channels(); ++j)
694 new_values(i, j) = !new_is_break(i) ? interp_amp(new_times(i), j, f_interval): 0.0;
695 }
696
697 p_times = new_times;
698 p_values = new_values;
699 p_is_val = new_is_break;
700 p_single_break = FALSE(1==0);
701 p_equal_space = TRUE(1==1);
702}

704float EST_Track::interp_amp(float x, int c, float fl)
705{
706 int i;
707 float x1, x2, y1, y2, m;
708
709 for (i = 0; i < num_frames(); ++i)
710 if ((p_times(i) + (fl / 2.0))> x)
711 break;
712
713 if (i == num_frames())
714 return p_values.a_no_check(i - 1,c);
715 if (i == 0)
716 return p_values.a_no_check(0, c);
717
718 if (track_break(i) && track_break(i - 1))
719 return 0.0;
720
721 if (track_break(i))
722 return p_values.a_no_check(i - 1, c);
723
724 else if (track_break(i - 1))
725 return p_values.a_no_check(i, c);
726
727 x1 = p_times(i - 1);
728 y1 = p_values.a_no_check(i - 1, c);
729 x2 = p_times(i);
730 y2 = p_values.a_no_check(i, c);
731
732 m = (y2 - y1) / (x2 -x1);
733 return ((x - x1) * m) + y1;
734}

736int EST_Track::interp_value(float x, float fl)
737{
738 int i;
739 int p, n;
740 float cf;
741
742 if (p_equal_space)
743 cf = shift();
744 else
745 cf = estimate_shift(x);
746
747 for (i = 0; i < num_frames(); ++i)
748 if ((p_times(i) + (fl / 2.0))> x)
749 break;
750 // This was:
751 // for (i = 0; i < num_frames(); ++i)
752 // if (p_times[i] > x)
753 // break;
754
755 if (i == 0) // must be a break for the first value. (can't have i -1).
756 return FALSE(1==0);
757
758 if ((!track_break(i)) && (!track_break(i -1)))
759 return TRUE(1==1);
760
761 p = prev_non_break(i);
762 n = next_non_break(i);
763
764 if ((x < p_times(p) + (cf / 2.0)) || (x > p_times(n) - (cf / 2.0)))
765 return TRUE(1==1); // rounding at edges
766
767 return FALSE(1==0);
768}
769
770float EST_Track::estimate_shift(float x)
771{
772 int i, j;
773 for (j = 0; j < num_frames(); ++j)
774 if (p_times(j) > x)
775 break;
776
777 for (i = j; i > 0; --i)
778 if ((!track_break(i)) && (!track_break(i - 1)))
779 return p_times(i) - p_times(i - 1);
780
781 for (i = j; i < num_frames() - 1; ++i)
782 if ((!track_break(i)) && (!track_break(i + 1)))
783 return p_times(i + 1) - p_times(i);
784
785 return 5.0; // default value
786}

812void EST_Track::rm_excess_breaks()
813{
814 int i, j, k;
815 EST_FVector new_times;
816 EST_CVector new_is_break;
817 EST_FMatrix new_values;
818
819 new_values.resize(num_channels(), num_frames());
820 new_times.resize(num_frames());
821 new_is_break.resize(num_frames());
822
823 for (i = 0; track_break(i); ++i); //rm leading breaks
824
825 for (j = 0; i < num_frames(); ++i, ++j)
826 {
827 for (k = 0; k < num_channels(); ++k)
828 new_values(j, k) = p_values.a_no_check(i, k);
829 new_times[j] = p_times(i);
830 new_is_break[j] = p_is_val(i);
831 while ((!new_is_break(j)) && (!val(i + 1)))
832 ++i;
833 }
834 p_times = new_times;
835 p_values = new_values;
836 p_is_val = new_is_break;
837 for (--j; track_break(j); --j) // "rm" trailing breaks
838 ;
839 p_times.resize(num_frames());
840 p_values.resize(num_frames(), num_channels());
841 p_is_val.resize(num_frames());
842
843 p_single_break = TRUE(1==1);
844}

846void EST_Track::rm_trailing_breaks()
847{
848 if (num_frames() <=0 )
849 return;
850 int start, end;
851
852 for (start = 0; start < num_frames(); ++start)
853 if (!track_break(start))
854 break;
855
856 for(end=num_frames(); end>0; end--)
857 if (!track_break(end-1))
858 break;
859
860 if (start==0 && end==num_frames())
861 return;
862
863 for (int i=start, j = 0; i < end; ++i, ++j)
864 {
865 p_times[j] = p_times(i);
866 for (int k = 0; k < num_channels(); k++)
867 a_no_check(j, k) = a_no_check(i, k);
868 p_is_val[j] = p_is_val(i);
869 }
870
871 p_values.resize(end-start, EST_CURRENT, 1);
872
873 p_times.resize(num_frames());
874 p_is_val.resize(num_frames());
875}

877void EST_Track::add_trailing_breaks()
878{
879 int i, j, k;
880 EST_FVector new_times;
881 EST_FMatrix new_values;
882 int new_num = num_frames();
883
884 if (!track_break(0))
885 new_num++;
886 if (!track_break(num_frames() - 1))
887 new_num++;
888
889 if (new_num == num_frames()) /*ie trailing breaks already there */
890 return;
891
892 new_times.resize(new_num);
893 new_values.resize(num_channels(), new_num);
894
895 j = 0;
896 if (!track_break(j))
897 set_break(j);
898
899 for (i = 0; i < num_frames(); ++i, ++j)
900 {
901 new_times[j] = p_times(i);
902 for (k = 0; k < num_channels(); ++k)
903 new_values(j, k) = p_values.a_no_check(i, k);
904 }
905
906 if (!track_break(num_frames() - 1))
907 set_break(j);
908
909 p_times = new_times;
910 p_values = new_values;
911 p_times.resize(num_frames());
912 p_values.resize(num_frames(), num_channels());
913}

915void EST_Track::pad_breaks()
916{
917 if (!p_single_break)
918 return;
919
920 if (!p_equal_space)
921 EST_error(EST_error_where = __null), (*EST_error_func)("pad_breaks: Can only operate on fixed data\n");
922
923 EST_FVector new_times;
924 EST_FMatrix new_values;
925 EST_CVector new_is_break;
926 int i, j, k, n;
927
928 n = (int)(((end())/ shift()) + 1.0);
929 int s = int(start()/ shift());
930
931 for (i = 0; i < n; ++i)
932 {
933 new_times[i] = (float) (i * shift());
934 for (k = 0; k < num_channels(); ++k)
935 new_values(k, i) = 0.0;
936 new_is_break[i] = 0;
937 }
938
939 for (i = 0, j = s; j < n; ++i, ++j)
940 {
941 if (track_break(i))
942 {
943 for (; new_times(j) < p_times(i + 1); ++j);
944 --j;
945 }
946 else
947 {
948 new_is_break[j] = 1;
949 for (k = 0; k < num_channels(); ++k)
950 new_values(j, k) = p_values.a_no_check(i, k);
951 }
952 }
953 new_is_break[j] = 1;
954 for (k = 0; k < num_channels(); ++k)
955 new_values(j, k) = p_values.a_no_check(i, k);
956
957 p_times = new_times;
958 p_values = new_values;
959 p_is_val = new_is_break;
960
961 p_times.resize(num_frames());
962 p_is_val.resize(num_frames());
963 p_values.resize(num_frames(), num_channels());
964
965 p_single_break = FALSE(1==0);
966}
