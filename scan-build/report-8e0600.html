77int writable_file(char *filename)
78{
79 // Returns TRUE if this is afile is writable or creatable, FALSE
80 // otherwise
81 // Note this is *not* guaranteed to work, if the file doesn't
82 // exist the directory is checked if its writable but it can
83 // lie, esp. with ro file systems and NFS.
84
85 if (streq(filename,"-")(strcmp(filename,"-")==0))
86 return TRUE(1==1);
87 else if (access(filename,W_OK2) == 0)
88 return TRUE(1==1);
89 else if ((access(filename,F_OK0) == -1) && // doesn't exists
90 (access(EST_Pathname(filename).directory(),W_OK2) == 0))
91 return TRUE(1==1); // probably;
92 else
93 return FALSE(1==0);
94}

168// Carry out equivalent of Bourne shell basename command, i.e. strip of
169// leading path and optionally remove extension. It assumes directories
170// are separated by a forward "/". This wont work on deviant OSs.
171EST_String basename(EST_String full, EST_String ext)
172{
173 if (full.contains("/"))
174 {
175 full= full.after(full.index("/", -1));
176// full= full.after("/"); //- don't know why this was here
177 }
178
179 if (ext == "*")
180 {
181 if (full.contains("."))
182 full = full.before(".", -1); // everything apart from last extension
183 }
184 else if (ext == "?")
185 {
186 if (full.contains("."))
187 full = full.before("."); // everything up to first extension
188 }
189 else if (ext != "")
190 full = full.before(ext);
191
192 return full;
193}

195void strip_quotes(EST_String &s, const EST_String quote_char)
196{
197 // if s is has quote_char as first and/or last char, strip them
198 if (s == "")
199 return;
200
201 if (quote_char(0) == s(0))
202 s = s.after(0);
203 if (quote_char(0) == s(s.length()-1))
204 s = s.before((int)(s.length()-1));
205}
