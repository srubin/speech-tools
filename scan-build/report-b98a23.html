220static float do_summary(WNode &tree,WDataSet &ds,ostream *output)
221{
222 if (wgn_dataset.ftype(wgn_predictee) == wndt_cluster)

234WNode *wgn_build_tree(float &score)
235{
236 // Build init node and split it while reducing the impurity
237 WNode *top = new WNode();
238 int margin = 0;
239
240 wgn_set_up_data(top->get_data(),wgn_dataset,wgn_held_out,TRUE(1==1));
241
242 margin = 0;
243 wagon_split(margin,*top); // recursively split data;
244
245 if (wgn_held_out > 0)
246 {
247 wgn_set_up_data(top->get_data(),wgn_dataset,wgn_held_out,FALSE(1==0));
248 top->held_out_prune();
249 }
250
251 if (wgn_prune)
252 top->prune();
253
254 score = summary_results(*top,0);
255
256 return top;
257}

287{
288 // Test tree against data to get summary of results
289 EST_StrStr_KVL pairs;

292 EST_String predict,real;
293 WNode *pnode;

298 for (p=dataset.head(); p != 0; p=p->next())

300 pnode = tree.predict_node((*dataset(p)));

302 if (wgn_count_field == -1)
303 count = 1.0;
304 else
305 count = dataset(p)->get_flt_val(wgn_count_field);
306 prob = pnode->get_impurity().pd().probability(predict);
307 H += (log(prob))*count;
308 type = dataset.ftype(wgn_predictee);
309 real = wgn_discretes[type].name(dataset(p)->get_int_val(wgn_predictee));
310 if (real == predict)
311 correct += count;
312 total += count;
313 pairs.add_item(real,predict,1);
314 }
315 for (i=0; i<wgn_discretes[dataset.ftype(wgn_predictee)].length(); i++)
316 lex.append(wgn_discretes[dataset.ftype(wgn_predictee)].name(i));
317
318 const EST_FMatrix &m = confusion(pairs,lex);
319
320 if (output != NULL__null)
321 {
322 print_confusion(m,pairs,lex); // should be to output not stdout
323 *output << ";; entropy " << (-1*(H/total)) << " perplexity " <<
324 pow(2.0,(-1*(H/total))) << endl;
325 }
326
327 // Minus it so bigger is better
328 if (wgn_opt_param == "entropy")
329 return -pow(2.0,(-1*(H/total)));
330 else

595 << " Correlation is " << ftoString(cor,4,1)
596 << " Mean (abs) Error " << ftoString(e.mean(),4,1)
597 << " (" << ftoString(e.stddev(),4,1) << ")" << endl;
598
599 cout << "RMSE " << ftoString(sqrt(se.mean()),4,1)
600 << " Correlation is " << ftoString(cor,4,1)
601 << " Mean (abs) Error " << ftoString(e.mean(),4,1)
602 << " (" << ftoString(e.stddev(),4,1) << ")" << endl;
603 }
604
605 if (wgn_opt_param == "rmse")
606 return -sqrt(se.mean()); // * -1 so bigger is better
607 else
608 return cor; // should really be % variance, I think
609}

611static int wagon_split(int margin, WNode &node)
612{
613 // Split given node (if possible)
614 WQuestion q;
615 WNode *l,*r;
616
617 node.set_impurity(WImpurity(node.get_data()));
618 q = find_best_question(node.get_data());
619
620/* printf("q.score() %f impurity %f\n",
621 q.get_score(),
622 node.get_impurity().measure()); */
623
624 double impurity_measure = node.get_impurity().measure();
625 double question_score = q.get_score();
626
627 if ((question_score < WGN_HUGE_VAL1.0e20) &&
628 (question_score < impurity_measure))
629
630 {
631 // Ok its worth a split
632 l = new WNode();
633 r = new WNode();
634 wgn_find_split(q,node.get_data(),l->get_data(),r->get_data());
635 node.set_subnodes(l,r);
636 node.set_question(q);
637 if (wgn_verbose)
638 {
639 int i;
640 for (i=0; i < margin; i++)
641 cout << " ";
642 cout << q << endl;
643 }
644 margin++;
645 wagon_split(margin,*l);
646 margin++;
647 wagon_split(margin,*r);
648 margin--;
649 return TRUE(1==1);
650 }
651 else
652 {
653 if (wgn_verbose)
654 {
655 int i;
656 for (i=0; i < margin; i++)
657 cout << " ";
658 cout << "stopped samples: " << node.samples() << " impurity: "
659 << node.get_impurity() << endl;
660 }
661 margin--;
662 return FALSE(1==0);
663 }
664}

670
671 y.resize(q.get_yes());
672 n.resize(q.get_no());

675 if (q.ask(*ds(i)) == TRUE(1==1))
676 y[iy++] = ds(i);
677 else
678 n[in++] = ds(i);
679
680}

682static WQuestion find_best_question(WVectorVector &dset)
683{
684 // Ask all possible questions and find the best one
685 int i;
686 float bscore,tscore;
687 WQuestion test_ques, best_ques;
688
689 bscore = tscore = WGN_HUGE_VAL1.0e20;
690 best_ques.set_score(bscore);
691 // test each feature with each possible question
692 for (i=0;i < wgn_dataset.width(); i++)
693 {
694 if ((wgn_dataset.ignore(i) == TRUE(1==1)) ||
695 (i == wgn_predictee))
696 tscore = WGN_HUGE_VAL1.0e20; // ignore this feature this time
697 else if (wgn_dataset.ftype(i) == wndt_binary)
698 {
699 construct_binary_ques(i,test_ques);
700 tscore = wgn_score_question(test_ques,dset);
701 }
702 else if (wgn_dataset.ftype(i) == wndt_float)
703 {
704 tscore = construct_float_ques(i,test_ques,dset);
705 }
706 else if (wgn_dataset.ftype(i) == wndt_ignore)
707 tscore = WGN_HUGE_VAL1.0e20; // always ignore this feature
708#if 0
709 // This doesn't work reasonably
710 else if (wgn_csubset && (wgn_dataset.ftype(i) >= wndt_class))
711 {
712 wagon_error("subset selection temporarily deleted")(std::cerr << "subset selection temporarily deleted" <<
std::endl,exit(-1));
713 tscore = construct_class_ques_subset(i,test_ques,dset);
714 }
715#endif
716 else if (wgn_dataset.ftype(i) >= wndt_class)
717 tscore = construct_class_ques(i,test_ques,dset);
718 if (tscore < bscore)
719 {
720 best_ques = test_ques;
721 best_ques.set_score(tscore);
722 bscore = tscore;
723 }
724 }
725
726 return best_ques;
727}

856static float construct_float_ques(int feat,WQuestion &ques,WVectorVector &ds)
857{
858 // Find out a split of the range that gives the best score
859 // Naively does this by partitioning the range into float_range_split slots
860 float tscore,bscore = WGN_HUGE_VAL1.0e20;
861 int d, i;
862 float p;
863 WQuestion test_q;
864 float max,min,val,incr;
865
866 test_q.set_fp(feat);
867 test_q.set_oper(wnop_lessthan);
868 ques = test_q;
869
870 min = max = ds(0)->get_flt_val(feat); /* set up some value */
871 for (d=0; d < ds.n(); d++)
872 {
873 val = ds(d)->get_flt_val(feat);
874 if (val < min)
875 min = val;
876 else if (val > max)
877 max = val;
878 }
879 if (max == min) // we're pure
880 return WGN_HUGE_VAL1.0e20;
881 incr = (max-min)/wgn_float_range_split;
882 // so do float_range-1 splits
883 /* We calculate this based on the number splits, not the increments, */
884 /* becuase incr can be so small it doesn't increment p */
885 for (i=0,p=min+incr; i < wgn_float_range_split; i++,p += incr )
886 {
887 test_q.set_operand1(EST_Val(p));
888 tscore = wgn_score_question(test_q,ds);
889 if (tscore < bscore)
890 {
891 ques = test_q;
892 bscore = tscore;
893 }
894 }
895
896 return bscore;
897}

910static float score_question_set(WQuestion &q, WVectorVector &ds, int ignorenth)
911{
912 // score this question as a possible split by finding
913 // the sum of the impurities when ds is split with this question
914 WImpurity y,n;
915 int d, num_yes, num_no;
916 float count;
917 WVector *wv;
918
919 num_yes = num_no = 0;
920 for (d=0; d < ds.n(); d++)
921 {
922 if ((ignorenth < 2) ||
923 (d%ignorenth != ignorenth-1))
924 {
925 wv = ds(d);
926 if (wgn_count_field == -1)
927 count = 1.0;
928 else
929 count = (*wv)[wgn_count_field];
930
931 if (q.ask(*wv) == TRUE(1==1))
932 {
933 num_yes++;
934 y.cumulate((*wv)[wgn_predictee],count);
935 }
936 else
937 {
938 num_no++;
939 n.cumulate((*wv)[wgn_predictee],count);
940 }
941 }
942 }
943
944 q.set_yes(num_yes);
945 q.set_no(num_no);
946
947 int min_cluster;
948
949 if ((wgn_balance == 0.0) ||
950 (ds.n()/wgn_balance < wgn_min_cluster_size))
951 min_cluster = wgn_min_cluster_size;
952 else
953 min_cluster = (int)(ds.n()/wgn_balance);
954
955 if ((y.samples() < min_cluster) ||
956 (n.samples() < min_cluster))
957 return WGN_HUGE_VAL1.0e20;
958
959 float ym,nm,bm;
960 ym = y.measure();
961 nm = n.measure();
962 bm = ym + nm;
963
964/* cout << q << endl;
965 printf("test question y %f n %f b %f\n",
966 ym, nm, bm); */
967
968 return bm/2.0;
969}

971float wgn_score_question(WQuestion &q, WVectorVector &ds)
972{
973 // This level of indirection was introduced for later expansion
974
975 return score_question_set(q,ds,1);
976}

978WNode *wagon_stepwise(float limit)
979{
980 // Find the best single features and incrementally add features
981 // that best improve result until it doesn't improve.
982 // This is basically to automate what Kurt was doing in building
983 // trees, he then automated it in PERL and as it seemed to work
984 // I put it into wagon itself.
985 // This can be pretty computationally intensive.
986 WNode *best = 0,*new_best = 0;
987 float bscore,best_score = -WGN_HUGE_VAL1.0e20;
988 int best_feat,i;
989 int nf = 1;
990
991 // Set all features to ignore
992 for (i=0; i < wgn_dataset.width(); i++)
993 wgn_dataset.set_ignore(i,TRUE(1==1));
994
995 for (i=0; i < wgn_dataset.width(); i++)
996 {
997 if ((wgn_dataset.ftype(i) == wndt_ignore) || (i == wgn_predictee))
998 {
999 // This skips the round not because this has anything to
1000 // do with this feature being (user specified) ignored
1001 // but because it indicates there is one less cycle that is
1002 // necessary
1003 continue;
1004 }
1005 new_best = wagon_stepwise_find_next_best(bscore,best_feat);
1006
1007 if ((bscore - fabs(bscore * (limit/100))) <= best_score)
1008 {
1009 // gone as far as we can
1010 delete new_best;
1011 break;
1012 }
1013 else
1014 {
1015 best_score = bscore;
1016 delete best;
1017 best = new_best;
1018 wgn_dataset.set_ignore(best_feat,FALSE(1==0));
1019 if (!wgn_quiet)
1020 {
1021 fprintf(stdoutstdout,"FEATURE %d %s: %2.4f\n",
1022 nf,
1023 (const char *)wgn_dataset.feat_name(best_feat),
1024 best_score);
1025 fflush(stdoutstdout);
1026 nf++;
1027 }
1028 }
1029 }
1030
1031 return best;
1032}

1034static WNode *wagon_stepwise_find_next_best(float &bscore,int &best_feat)
1035{
1036 // Find which of the currently ignored features will best improve
1037 // the result
1038 WNode *best = 0;
1039 float best_score = -WGN_HUGE_VAL1.0e20;
1040 int best_new_feat = -1;
1041 int i;
1042
1043 for (i=0; i < wgn_dataset.width(); i++)
1044 {
1045 if (wgn_dataset.ftype(i) == wndt_ignore)
1046 continue; // user wants me to ignore this completely
1047 else if (i == wgn_predictee) // can't use the answer
1048 continue;
1049 else if (wgn_dataset.ignore(i) == TRUE(1==1))
1050 {
1051 WNode *current;
1052 float score;
1053
1054 // Allow this feature to participate
1055 wgn_dataset.set_ignore(i,FALSE(1==0));
1056
1057 current = wgn_build_tree(score);
1058
1059 if (score > best_score)
1060 {
1061 best_score = score;
1062 delete best;
1063 best = current;
1064 best_new_feat = i;
1065// fprintf(stdout,"BETTER FEATURE %d %s: %2.4f\n",
1066// i,
1067// (const char *)wgn_dataset.feat_name(i),
1068// best_score);
1069// fflush(stdout);
1070 }
1071 else
1072 delete current;
1073
1074 // switch it off again
1075 wgn_dataset.set_ignore(i,TRUE(1==1));
1076 }
1077 }
1078
1079 bscore = best_score;
1080 best_feat = best_new_feat;
1081 return best;
