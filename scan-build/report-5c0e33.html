401int relation_divide(EST_RelationList &slist, EST_Relation &lab,
402 EST_Relation &keylab,
403 EST_StrList &blank, EST_String ext)
404{ // divides a single relation into multiple chunks according to the
405 // keylab relation. If the keylab boundary falls in the middle of a label,
406 // the label is assigned to the chunk which has the most overlap with
407 // it. Some labels may be specified in the "blank" list which means thy
408 // are duplicated across boundaries.
409
410 EST_Relation a, newkey;
411 EST_Item *s, *k, *t = 0, *n;
412 EST_String filename;
413 float kstart;
414
415 slist.clear();
416
417 if ((keylab.tail())->F("end") < (lab.tail())->F("end"))
418 {
419 cerr << "Key file must extend beyond end of label file\n";
420 return -1;
421 }
422
423 // find a the first keylab that will make a non-empty file
424 for (k = keylab.head(); k ; k = k->next())
425 if (k->F("end") > lab.head()->F("end"))
426 break;
427 if (k != NULL__null)
428 filename = (EST_String)k->f("file");
429 else
430 filename = "no_name";
431 a.f.set("name", (filename + ext));
432 kstart = 0.0;
433
434 for (s = lab.head(); s; s = s->next())
435 {
436 n = s->next();
437 if (n == 0)
438 {
439 t = a.append(s);
440 t->set("end", (s->F("end") - kstart));
441 break;
442 }
443 if (n->F("end") > k->F("end"))
444 {
445 if (((n->F("end") - k->F("end")) <
446 (k->F("end") - start(n))) ||
447 is_in_class(n->name(), blank))
448 {
449 a.append(s);
450 t->set("end", (s->F("end") - kstart));
451
452 t = a.append(n);
453 t->set("end", (k->F("end") - kstart));
454
455 if (!is_in_class(n->name(), blank))
456 s = s->next();
457 }
458 else
459 {
460 t = a.append(s);
461 t->set("end", (k->F("end") - kstart));
462 }
463
464 slist.append(a);
465 k = k->next();
466 kstart = start(k);
467 a.clear();
468 filename = (EST_String)k->f("file");
469 a.f.set("name", (filename + ext));
470 }
471 else
472 {
473 t = a.append(s);
474 t->set("end", (s->F("end") - kstart));
475 }
476 }
477 slist.append(a);
478
479 return 0;
480}

528void map_match_times(EST_Relation &target, const EST_String &match_name,
529 const EST_String &time_name, bool do_start)
530{
531 EST_Item *s, *t, *p;
532 float prev_end, inc, first_end, last_end;
533 int i;
534
535 // first pass, copy times as appropriate, and find first
536 // and last defined ends
537 // This is hacky and certainly won't work for many cases
538
539 first_end = -1.0;
540 prev_end = 0.0;
541 last_end = 0.0;
542
543// cout << "surface: " << surface << endl;
544
545 for (s = target.head(); s; s = s->next())
546 {
547 if ((t = daughter1(s->as_relation(match_name))) != 0)
548 {
549 s->set(time_name + "end", t->F("end"));
550 if (do_start)
551 s->set(time_name + "start", t->F("start"));
552
553 last_end = t->F("end");
554 if (first_end < 0.0)
555 first_end = t->F("end");
556 }
557 }
558
559 if (!target.head()->f_present(time_name + "end"))
560 {
561 target.head()->set(time_name + "end", first_end / 2.0);
562 if (do_start)
563 target.head()->set(time_name + "start", 0.0);
564 }
565
566 if (!target.tail()->f_present(time_name + "end"))
567 {
568 target.tail()->set(time_name + "end", last_end + 0.01);
569 if (do_start)
570 target.tail()->set(time_name + "start", last_end);
571 }
572
573 for (s = target.head(); s; s = s->next())
574 {
575 if (!s->f_present(time_name + "end"))
576 {
577// cout << "missing end feature for " << *s << endl;
578 for (i = 1, p = s; p; p = p->next(), ++i)
579 if (p->f_present(time_name + "end"))
580 break;
581 inc = (p->F(time_name + "end") - prev_end) / ((float) i);
582// cout << "inc is : " << inc << endl;
583
584// cout << "stop phone is " << *p << endl;
585
586 for (i = 1; s !=p ; s = s->next(), ++i)
587 {
588 s->set(time_name + "end", (prev_end + ((float) i * inc)));
589 if (do_start)
590 s->set(time_name + "start", (prev_end+((float) (i - 1 )* inc)));
591 }
592 }
593 prev_end = s->F("end");
594 }

631void relation_convert(EST_Relation &lab, EST_Option &al, EST_Option &op)
632{
633 if (al.present("-shift"))
634 shift_label(lab, al.fval("-shift"));
635
636 // fix option later.
637 if (al.present("-extend"))
638 al.override_fval("-length",
639 al.fval("-extend",0) * lab.tail()->F("end"));
640
641 // quantize (ie round up or down) label times
642 if (al.present("-q"))
643 quantize(lab, al.fval("-q"));
644
645 if (al.present("-start"))
646 {
647 if (!al.present("-end"))
648 cerr << "-start option must be used with -end option\n";
649 else
650 extract(lab, al.fval("-start"), al.fval("-end"), lab);
651 }
652
653 if (al.present("-class"))
654 convert_to_broad_class(lab, al.val("-class"), op);
655
656 else if (al.present("-pos"))
657 {
658 EST_StrList bclass;
659 StringtoStrList(al.val("-lablist"), bclass);
660 convert_to_broad(lab, bclass);
661 }
662 else if (al.present("-sed"))
663 edit_labels(lab, al.val("-sed"));
664 else if (al.present("-map"))
665 {
666 EST_Option map;
667 if (map.load(al.val("-map")) != format_okread_ok)
668 return;
669 label_map(lab, map);
670 }
671}
