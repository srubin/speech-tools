291void lpc2cep(const EST_FVector &lpc, EST_FVector &cep)
292{
293 int n, k;
294 float sum;
295 int order = lpc.length() - 1;
296
297 for (n = 1; n <= order && n <= cep.length(); n++)
298 {
299 sum = 0.0;
300 for (k = 1; k < n; k++)
301 sum += k * cep(k-1) * lpc(n - k);
302 cep[n-1] = lpc(n) + sum / n;
303 }
304
305 /* be wary of these interpolated values */
306 for(n = order + 1; n <= cep.length(); n++)
307 {
308 sum = 0.0;
309 for (k = n - (order - 1); k < n; k++)
310 sum += k * cep(k-1) * lpc(n - k);
311 cep[n-1] = sum / n;
312 }
313
314 /* very occasionally the above can go unstable, fudge if this happens */
315
316 for (n = 0; n < cep.length(); n++)
317 {
318 // check if NaN -- happens on some frames of silence
319 if (isnanf(cep[n])__isnanf(cep[n]) ) cep[n] = 0.0;
320
321 if (cep[n] > MAX_ABS_CEPS4.0){
322 cerr << "WARNING : cepstral coeff " << n << " was " <<
323 cep[n] << endl;
324 cerr << "lpc coeff " << n << " = " << lpc(n + 1) << endl;
325
326 cep[n] = MAX_ABS_CEPS4.0;
327 }
328 if (cep[n] < -MAX_ABS_CEPS4.0) {
329 cerr << "WARNING : cepstral coeff " << n << " was " <<
330 cep[n] << endl;
331 cep[n] = -MAX_ABS_CEPS4.0;
332 }
333 }

392void ref2lpc(const EST_FVector &ref, EST_FVector &lpc)
393{
394 // Here we use Christopher Longet Higgin's algorithm converted to
395 // an equivalent by awb. It doesn't have the reverse order or
396 // negation requirement.
397 int order = ref.length() - 1;
398 float a, b;
399 int n, k;
400
401 for (n=0; n < order; n++)
402 {
403 lpc[n] = ref(n);
404 for (k=0; 2 * (k+1) <= n + 1; k++)
405 {
406 a = lpc[k];
407 b = lpc[n-(k + 1)];
408 lpc[k] = a-b * lpc[n];
409 lpc[n-(k+1)] = b-a * lpc[n];
410 }
411 }
412}

434void sig2lpc(const EST_FVector &sig, EST_FVector &acf,
435 EST_FVector &ref, EST_FVector &lpc)
436{
437
438 int i, j;
439 float e, ci, sum;
440 int order = lpc.length() -1;
441 EST_FVector tmp(order);
442 int stableorder=-1;
443
444 if ((acf.length() != ref.length()) ||
445 (acf.length() != lpc.length()))
446 EST_error(EST_error_where = __null), (*EST_error_func)("sig2lpc: acf, ref are not of lpc's order");
447
448 //cerr << "sig2lpc order " << order << endl;
449
450
451 for (i = 0; i <= order; i++)
452 {
453 sum = 0.0;
454 for(j = 0; j < sig.length() - i; j++)
455 sum += sig.a_no_check(j) * sig.a_no_check(j + i);
456 acf.a_no_check(i) = sum;
457 }
458
459 // find lpc coefficients
460 e = acf.a_no_check(0);
461 lpc.a_no_check(0) = 1.0;
462
463 for (i = 1; i <= order; i++)
464 {
465 ci = 0.0;
466 for(j = 1; j < i; j++)
467 ci += lpc.a_no_check(j) * acf.a_no_check(i-j);
468 if (e == 0)
469 ref.a_no_check(i) = ci = 0.0;
470 else
471 ref.a_no_check(i) = ci = (acf.a_no_check(i) - ci) / e;
472 //Check stability of the recursion
473 if (absval(ci) < 1.000000)
474 {
475 lpc.a_no_check(i) = ci;
476 for (j = 1; j < i; j++)
477 tmp.a_no_check(j) = lpc.a_no_check(j) -
478 (ci * lpc.a_no_check(i-j));
479 for( j = 1; j < i; j++)
480 lpc.a_no_check(j) = tmp.a_no_check(j);
481
482 e = (1 - ci * ci) * e;
483 stableorder = i;
484 }
485 else break;
486 }
487 if (stableorder != order)
488 {
489 fprintf(stderrstderr,
490 "warning:levinson instability, order restricted to %d\n",
491 stableorder);
492 for (; i <= order; i++)
493 lpc.a_no_check(i) = 0.0;
494 }
495
496 // normalisation for frame length
497 lpc.a_no_check(0) = e / sig.length();
498}

516float lpredict2(EST_FVector &adc, int wsize,
517 EST_FVector &acf, float *ref, float *lpc,
518 int order)
519{
520 int i, j;
521 float e, ci, sum;
522 EST_TBuffer<float> tmp(order);
523 int stableorder=-1;
524
525 EST_FVector vref(order + 1), vlpc(order + 1);
526
527 for (i = 0; i <= order; i++)
528 {
529 sum = 0.0;
530 for (j = 0; j < wsize - i; j++)
531 sum += adc[j] * adc[j + i];
532 acf[i] = sum;
533 }
534 /* find lpc coefficients */
535 e = acf[0];
536 lpc[0] = 1.0;
537 for(i = 1; i <= order; i++) {
538 ci = 0.0;
539 for(j = 1; j < i; j++)
540 ci += lpc[j] * acf[i-j];
541 ref[i] = ci = (acf[i] - ci) / e;
542 //Check stability of the recursion
543 if (absval(ci) < 1.000000) {
544 lpc[i] = ci;
545 for(j = 1; j < i; j++)
546 tmp[j] = lpc[j] - ci * lpc[i-j];
547 for(j = 1; j < i; j++)
548 lpc[j] = tmp[j];
549 e = (1 - ci * ci) * e;
550 stableorder = i;
551 }
552 else break;
553 }
554 if (stableorder != order) {
555 fprintf(stderrstderr,
556 "warning:levinson instability, order restricted to %d\n",
557 stableorder);
558 for (;i<=order;i++)
559 lpc[i]=0.0;
560 }
561 return(e);
562}

567 EST_FVector &fbank_frame,
568 const float sample_rate,

570 const bool take_log)

572

576
577 // upper and lower limits of filter bank
578 // where the upper limit depends on the sampling frequency
579 // TO DO : add low/high pass filtering HERE
580 float mel_low=0;
581 float mel_high=Hz2Mel(sample_rate / 2);
582
583 // FFT this frame. FFT order will be computed by sig2fft
584 // FFT frame returned will be half length of actual FFT performed
585 sig2fft(sig,fft_frame,use_power_rather_than_energy);
586
587 // this is more easily understood as half the sampling
588 // frequency over half the fft order, but fft_frame_length()
589 // is already halved
590 Hz_per_fft_coeff = 0.5 * sample_rate / fft_frame.length();
591
592 fbank_order = fbank_frame.length();
593
594 // store the list of centre frequencies and lower and upper bounds of

596 EST_FVector mel_fbank_centre_frequencies(fbank_order+2);
597
598 mel_fbank_centre_frequencies[0]=mel_low;
599
600 for(i=1;i<=fbank_order;i++)
601 mel_fbank_centre_frequencies[i] = mel_low +
602 (float)(i) * (mel_high - mel_low) / (fbank_order+1);
603
604 mel_fbank_centre_frequencies[fbank_order+1]=mel_high;
605
606 // bin FFT in Mel filters
607 fft2fbank(fft_frame,
608 fbank_frame,
609 Hz_per_fft_coeff,
610 mel_fbank_centre_frequencies);
611
612 if(take_log)
613 for(i=0;i<fbank_frame.length();i++)
614 fbank_frame[i] = safe_log(fbank_frame[i]);
615
616}

618void sig2fft(const EST_FVector &sig,
619 EST_FVector &fft_vec,
620 const bool use_power_rather_than_energy)
621{
622
623 int i,half_fft_order;
624 float real,imag;
625 float window_size = sig.length();
626 int fft_order; /* = fft_vec.length();*/
627
628 // work out FFT order required
629 fft_order = 2;
630 while (window_size > fft_order)
631 fft_order *= 2;
632
633 fft_vec = sig;
634
635 // pad with zeros
636 fft_vec.resize(fft_order);
637
638 // in place FFT
639 (void)fastFFT(fft_vec);
640
641 // of course, we only need the lower half of the fft
642 half_fft_order = fft_order/2;
643
644 for(i=0;i<half_fft_order;i++)
645 {
646 real = fft_vec(i*2);
647 imag = fft_vec(i*2 + 1);
648
649 fft_vec[i] = real * real + imag * imag;
650
651 if(!use_power_rather_than_energy)
652 fft_vec[i] = sqrt(fft_vec(i));
653
654 }
655
656 // discard mirror image, retaining energy/power spectrum
657 fft_vec.resize(half_fft_order);
658
659}

664 EST_FVector &fbank_vec,
665 const float Hz_per_fft_coeff,
666 const EST_FVector &mel_fbank_frequencies)
667{
668
669 // expects "half length" FFT - i.e. energy or power spectrum
670 // energy is magnitude; power is squared magnitude
671
672 // mel_fbank_frequencies is a vector of centre frequencies
673 // BUT : first element is lower bound of first filter
674 // last element is upper bound of final filter
675 // i.e. length = num filters + 2
676

678 float this_mel_centre,this_mel_low,this_mel_high;
679 EST_FVector filter;

681
682 // check that fbank_vec and mel_fbank_frequencies lengths match
683 if(mel_fbank_frequencies.length() != fbank_vec.length() + 2)
684 {
685 EST_error(EST_error_where = __null), (*EST_error_func)("Filter centre frequencies length (%i) is not equal to fbank order (%i) plus 2\n",mel_fbank_frequencies.length(),
686 fbank_vec.length());
687 return;
688 }
689
690 // filters are computed on the fly
691 for(i=0;i<fbank_vec.length();i++)
692 {
693
694 // work out shape of the i'th filter
695 this_mel_low=mel_fbank_frequencies(i);
696 this_mel_centre=mel_fbank_frequencies(i+1);
697 this_mel_high=mel_fbank_frequencies(i+2);
698
699 make_mel_triangular_filter(this_mel_centre,this_mel_low,this_mel_high,
700 Hz_per_fft_coeff,fft_frame.length(),
701 fft_index_start,filter);
702
703 // do filtering
704 fbank_vec[i]=0.0;
705 for(k=0;k<filter.length();k++)
706 fbank_vec[i] += fft_frame(fft_index_start + k) * filter(k);
707 }
708
709}

712void fbank2melcep(const EST_FVector &fbank_vec,
713 EST_FVector &mfcc_vec,
714 const float liftering_parameter,
715 const bool include_c0)
716{
717 // a cosine transform of the fbank output
718 // remember to pass LOG fbank params (energy or power)
719
720 int i,j,actual_mfcc_index;
721 float pi_i_over_N,cos_xform_order,const_factor;
722 float PI_over_liftering_parameter;
723
724 if(liftering_parameter != 0.0)
725 PI_over_liftering_parameter = PI3.14159265358979323846 / liftering_parameter;
726 else
727 PI_over_liftering_parameter = PI3.14159265358979323846; // since sin(n.PI) == 0
728
729 // if we are not including cepstral coeff 0 (c0) then we need
730 // to do a cosine transform 1 longer than otherwise
731 cos_xform_order = include_c0 ? mfcc_vec.length() : mfcc_vec.length() + 1;
Value stored to 'cos_xform_order' is never read
732
733 const_factor = sqrt(2 / (float)(fbank_vec.length()));
734
735 for(i=0;i<mfcc_vec.length();i++)
736 {
737
738 actual_mfcc_index = include_c0 ? i : i+1;
739
740 pi_i_over_N =
741 PI3.14159265358979323846 * (float)(actual_mfcc_index) / (float)(fbank_vec.length());
742
743 for(j=0;j<fbank_vec.length();j++)
744 // j + 0.5 is because we want (j+1) - 0.5
745 mfcc_vec[i] += fbank_vec(j) * cos(pi_i_over_N * ((float)j + 0.5));
746
747 mfcc_vec[i] *= const_factor;
748
749 // liftering
750 mfcc_vec[i] *= 1 + (0.5 * liftering_parameter
751 * sin(PI_over_liftering_parameter * (float)(actual_mfcc_index)));
752 }
753
754}

757void make_mel_triangular_filter(const float this_mel_centre,
758 const float this_mel_low,
759 const float this_mel_high,
760 const float Hz_per_fft_coeff,
761 const int half_fft_order,
762 int &fft_index_start,
763 EST_FVector &filter)
764{
765
766 // makes a triangular (on a Mel scale) filter and creates
767 // a weight vector to apply to FFT coefficients
768
769 int i,filter_vector_length,fft_index_stop;
770 float rise_slope,fall_slope,this_mel;
771
772
773 // slopes are in units per Mel
774 // this is important - slope is linear in MEl domain, not Hz
775 rise_slope = 1/(this_mel_centre - this_mel_low);
776 fall_slope = 1/(this_mel_centre - this_mel_high);
777
778
779 // care with rounding - we want FFT indices **guaranteed**
780 // to be within filter so we get no negative filter gains
781 // (irint gives the _nearest_ integer)
782
783 // round up
784 if(this_mel_low == 0)
785 fft_index_start=0;
786 else
787 fft_index_start = irint(0.5 + (Mel2Hz(this_mel_low) / Hz_per_fft_coeff));
788
789 // round down
790 fft_index_stop = irint((Mel2Hz(this_mel_high) / Hz_per_fft_coeff) - 0.5);
791
792 if(fft_index_stop > half_fft_order-1)
793 fft_index_stop = half_fft_order-1;
794
795
796 filter_vector_length = fft_index_stop - fft_index_start + 1;
797 filter.resize(filter_vector_length);
798
799 for(i=0;i<filter_vector_length;i++)
800 {
801 this_mel = Hz2Mel( (i + fft_index_start) * Hz_per_fft_coeff );
802
803 if(this_mel <= this_mel_centre)
804 {
805 filter[i] = rise_slope * (this_mel - this_mel_low);
806 }
807 else
808 {
809 filter[i] = 1 + fall_slope * (this_mel - this_mel_centre);
810 }
811
812 }
813
814}

817float Hz2Mel(float frequency_in_Hertz)
818{
819 return 1127 * log(1 + frequency_in_Hertz/700.0);
820}

822float Mel2Hz(float frequency_in_Mel)
823{
824 return (exp(frequency_in_Mel / 1127) - 1) * 700;
