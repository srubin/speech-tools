220double EST_SCFG_Chart::find_best_tree_cal(int start,int end,int p)
221{
222 // Find the best parse for non/terminal p between start and end
223 int best_j = -1;
224 int best_q = -1, best_r = -1;
225 double best_prob = 0;
226
227 if (end-1 == start)
228 {
229 best_prob = grammar->prob_U(p,wfst[start]->d1());
230 if (best_prob > 0)
231 edges[start][end][p] =
232 new EST_SCFG_Chart_Edge(best_prob*wfst[start]->prob(),
233 wfst[start]->d1(),0,-1);
234 else
235 edges[start][end][p] = emptyedge;
236 return best_prob;
237 }
238 else
239 {
240 // for all rules expanding p find total and best prob
241 double s=0,t_prob,left,right;
242 int j,q,r;
243 int nt = grammar->num_nonterminals();
244
245 for (q=0; q < nt; q++)
246 for (r=0; r < nt; r++)
247 {
248 double pBpqr = grammar->prob_B(p,q,r);
249 if (pBpqr > 0)
250 {
251 for (j=start+1; j < end; j++)
252 {
253 left=find_best_tree(start,j,q);
254 if (left > 0)
255 {
256 right = find_best_tree(j,end,r);
257 t_prob = pBpqr * left * right;
258 if (t_prob > best_prob)
259 {
260 best_prob = t_prob;
261 best_q = q;
262 best_r = r;
263 best_j = j;
264 }
265 s += t_prob;
266 }
267 }
268 }
269 }
270
271 if (best_prob > 0)
272 edges[start][end][p] =
273 new EST_SCFG_Chart_Edge(s,best_q,best_r,best_j);
274 else
275 edges[start][end][p] = emptyedge;
276 return s;
277 }
