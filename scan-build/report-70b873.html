83/* First you must realise there is in fact a number of very similar but */
84/* subtly different header formats that appear on ESPS files. */
85/* ESPS_FEA and ESPS_SD (others for filters, spectrograms, etc) */
86/* The basic format is */
87/* preamble */
88/* fixed header */
89/* variable header */
90/* field descriptions (type and dimensions) */
91/* field names */
92/* header FEAs (maybe with values) */
93/* old and foreign headers (mostly ignored here) */
94/* data records themselves */

96/* esps_fea contain a name and possibly a value. They appear in the */
97/* variable part of the header */
98esps_fea new_esps_fea()
99{
100 esps_fea r = walloc(struct ESPS_FEA_struct,1)((struct ESPS_FEA_struct *)safe_walloc(sizeof(struct ESPS_FEA_struct
)*(1)));
101 r->type = 0;
102 r->clength = 0;
103 r->name = NULL__null;
104 r->dtype = 0;
105 r->count = 0;
106 r->v.ival = NULL__null;
107 return r;
108}

153{
154 /* Add a new field to the record */
155 char **names = hdr->field_name;
156 short *types = hdr->field_type;
157 int *dims = hdr->field_dimension;
158 int i;
159
160 hdr->field_name = walloc(char *,hdr->num_fields+1)((char * *)safe_walloc(sizeof(char *)*(hdr->num_fields+1))
);
161 hdr->field_type = walloc(short,hdr->num_fields+1)((short *)safe_walloc(sizeof(short)*(hdr->num_fields+1)));
162 hdr->field_dimension = walloc(int,hdr->num_fields+1)((int *)safe_walloc(sizeof(int)*(hdr->num_fields+1)));

165 hdr->field_name[i] = names[i];
166 hdr->field_type[i] = types[i];

170 wfree(types);

448static int esps_alloc_fea(esps_fea r)
449{
450 switch (r->dtype)
451 {
452 case 0: /* nothing */
453 break;
454 case ESPS_DOUBLE1:
455 r->v.dval = walloc(double,r->count)((double *)safe_walloc(sizeof(double)*(r->count))); break;
456 case ESPS_FLOAT2:
457 r->v.fval = walloc(float,r->count)((float *)safe_walloc(sizeof(float)*(r->count))); break;
458 case ESPS_INT3:

462 case ESPS_CHAR5:

464 default:

466 return -1;
467 }

469 return 0;

472void write_esps_fea(FILE *fd, esps_fea t,esps_hdr hdr)
473{
474 /* write out this fea */
475 (void)hdr;
476 short clength;
477 char *nspace;
478 int i;
479
480 fwrite(&t->type,2,1,fd);
481 clength = (strlen(t->name)+3)/4;
482 fwrite(&clength,2,1,fd);
483 nspace = walloc(char, clength*4)((char *)safe_walloc(sizeof(char)*(clength*4)));
484 memset(nspace,0,clength*4);
485 memmove(nspace,t->name,strlen(t->name));
486 fwrite(nspace,1,clength*4,fd);
487 wfree(nspace);
488 if ((t->type == 11) ||
489 (t->type == 1) ||
490 (t->type == 15))
491 return;
492 fwrite(&t->count,4,1,fd);
493 fwrite(&t->dtype,2,1,fd);
494
495 for (i=0; i<t->count; i++)
496 {
497 switch(t->dtype)
498 {
499 case ESPS_DOUBLE1:
500 fwrite(&t->v.dval[i],8,1,fd); break;
501 case ESPS_FLOAT2:
502 fwrite(&t->v.fval[i],4,1,fd); break;
503 case ESPS_INT3:
504 fwrite(&t->v.ival[i],4,1,fd); break;
505 case ESPS_SHORT4:
506 fwrite(&t->v.sval[i],2,1,fd); break;
507 case ESPS_CHAR5:
508 fwrite(&t->v.cval[i],1,1,fd); break;
509 default:
510 fprintf(stderrstderr,"ESPS write_hdr: unsupported FEA dtype %d\n",
511 t->dtype);
512 }
513 }
514 return;
515}

554esps_fea read_esps_fea(FILE *fd, esps_hdr hdr)
555{
556 /* read next FEA record at point */
557 esps_fea r = new_esps_fea();
558 short sdata;
559 int i;
560 int idata;
561 float fdata;
562 double ddata;
563 char cdata;
564
565 fread(&sdata,2,1,fd);
566 if (hdr->swapped) sdata = SWAPSHORT(sdata)((((unsigned)sdata) & 0xff) << 8 | (((unsigned)sdata
) & 0xff00) >> 8);
567 r->type = sdata;
568 if (r->type == 0) /* a field name */
569 { /* next short is the size in bytes */
570 fread(&sdata,2,1,fd);
571 if (hdr->swapped) sdata = SWAPSHORT(sdata)((((unsigned)sdata) & 0xff) << 8 | (((unsigned)sdata
) & 0xff00) >> 8);
572 r->clength = sdata;
573 }
574 else if ((r->type == 13) || /* a feature and value */
575 (r->type == 11) || /* a single string (comment ?) */
576 (r->type == 1) || /* a filename */
577 (r->type == 4) || /* a filename */
578 (r->type == 15)) /* directory name */
579 {
580 fread(&sdata,2,1,fd);
581 if (hdr->swapped) sdata = SWAPSHORT(sdata)((((unsigned)sdata) & 0xff) << 8 | (((unsigned)sdata
) & 0xff00) >> 8);
582 r->clength = sdata * 4;
583 }
584 else
585 {
586 fprintf(stderrstderr,"ESPS: fea record unknown type\n");
587 wfree(r);
588 return NULL__null;
589 }
590 r->name = walloc(char,r->clength+1)((char *)safe_walloc(sizeof(char)*(r->clength+1)));
591 fread(r->name,1,r->clength,fd);
592 r->name[r->clength] = '\0';
593 if ((r->type == 11) || /* a single string */
594 (r->type == 1) || /* a filename */
595 (r->type == 15)) /* directory name */
596 return r;
597 fread(&idata,4,1,fd);
598 if (hdr->swapped) idata = SWAPINT(idata)((((unsigned)idata) & 0xff) << 24 | (((unsigned)idata
) & 0xff00) << 8 | (((unsigned)idata) & 0xff0000
) >> 8 | (((unsigned)idata) & 0xff000000) >> 24
);
599 r->count = idata;
600 fread(&sdata,2,1,fd);
601 if (hdr->swapped) sdata = SWAPSHORT(sdata)((((unsigned)sdata) & 0xff) << 8 | (((unsigned)sdata
) & 0xff00) >> 8);
602 r->dtype = sdata;
603 if (esps_alloc_fea(r) == -1)
604 return NULL__null;
605 for (i=0; i<r->count; i++)
606 {
607 switch (r->dtype)
608 {
609 case ESPS_DOUBLE1:
610 fread(&ddata,8,1,fd);
611 if (hdr->swapped) swapdouble(&ddata);
612 r->v.dval[i] = ddata;
613 break;
614 case ESPS_FLOAT2:
615 fread(&fdata,4,1,fd);
616 if (hdr->swapped) swapfloat(&fdata);
617 r->v.fval[i] = fdata;
618 break;
619 case ESPS_INT3:
620 fread(&idata,4,1,fd);
621 if (hdr->swapped) idata = SWAPINT(idata)((((unsigned)idata) & 0xff) << 24 | (((unsigned)idata
) & 0xff00) << 8 | (((unsigned)idata) & 0xff0000
) >> 8 | (((unsigned)idata) & 0xff000000) >> 24
);
622 r->v.ival[i] = idata;
623 break;
624 case ESPS_SHORT4:
625 fread(&sdata,2,1,fd);
626 if (hdr->swapped) sdata = SWAPSHORT(sdata)((((unsigned)sdata) & 0xff) << 8 | (((unsigned)sdata
) & 0xff00) >> 8);
627 r->v.sval[i] = sdata;
628 break;
629 case ESPS_CHAR5:
630 fread(&cdata,1,1,fd);
631 r->v.cval[i] = cdata;
632 break;
633 default:
634 fprintf(stderrstderr,"ESPS read_hdr: unsupported FEA dtype %d\n",r->dtype);
635 wfree(r);
636 return NULL__null;
637 }
638 }
639
640 return r;
641}

643static char *esps_get_field_name(FILE *fd, esps_hdr hdr, int expect_source)
644{
645 /* read the next field name */
646 short size=0; /* bet its really a short */
647 char *name;
648
649 if (fread(&size,2,1,fd) != 1)
650 {
651 fputs("error reading field name size\n", stderrstderr);
