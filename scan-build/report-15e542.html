62static int sorttest(const void *a, const void *b)
63{ // for use with qsort C library function.
64 float *c = (float *)a;
65 float *d = (float *)b;
66 float res = (*c - *d);
67 if (res == 0.0)
68 return 0;
69 return (res < 0.0) ? -1 : 1;
70}

108void simple_med_smooth(EST_Track &c, int n, int channel)
109{// simple median smoother of order n
110
111
112 // windows longer than twice the track length cause problems
113 // here is one solution
114 if(n > c.num_frames())
115 n=c.num_frames();
116
117 // and tiny windows don't work either
118 // can't do median of 2 of fewer points
119 if(n < 3)
120 return;
121
122 int i, j, h, k;
123 float *a = new float[c.num_frames()];
124 float *m = new float[n];
125 h = n/2;
126
127
128 // sort start using < n order smoothing
129 for (i = 0; i < h; ++i)
130 {
131 k = (i * 2) + 1;
132 for (j = 0; j < k; ++j)
133 m[j] = c.a(j, channel);
134 qsort(m, k, sizeof(float), sorttest);
135 a[i] = m[i];
136 }
137
138 // sort main section using n order smoothing
139 for (i = h; i < c.num_frames() - h; ++i)
140 {
141 for (j = 0; j < n; ++j)
142 m[j] = c.a(i - h + j, channel);
143
144 qsort(m, n, sizeof(float), sorttest);
145 a[i] = m[h];
146 }
147 // sort end section using < n order smoothing
148 for (; i < c.num_frames(); ++i)
149 {
150 k = ((c.num_frames() - i)* 2) -1;
151 for (j = 0; j < k; ++j)
152 m[j] = c.a(i - (k/2) + j, channel);
153 qsort(m, k, sizeof(float), sorttest);
154 a[i] = m[k/2];
155 }
156
157 for (i = 0; i < c.num_frames(); ++i)
158 c.a(i,channel) = a[i];
159
160 delete [] a;
161 delete [] m;
162}

164void simple_mean_smooth(EST_Track &c, int n, int channel)
165{ // simple mean smoother of order n
166 int i, j, h, k=1;
167 float *a = new float[c.num_frames()];
168 float sum;
169 h = n/2;
170
171 for (i = 0; i < h; ++i)
172 {
173 k = (i * 2) + 1;
174 sum = 0.0;
175 for (j = 0; j < k; ++j)
176 sum += c.a(j, channel);
177 a[i] = sum /(float) k;
178 }
179
180 k= h*2 + 1;
181
182 for (i = h; i < c.num_frames() - h; ++i)
183 {
184 sum = 0.0;
185 for (j = 0; j < k; ++j)
186 sum += c.a(i - h + j, channel);
187 a[i] = sum /(float) k;
188 }
189
190 for (; i < c.num_frames(); ++i)
191 {
192 k = ((c.num_frames() - i)* 2) -1;
193 sum = 0.0;
194 for (j = 0; j < k; ++j)
195 sum += c.a(i - (k/2) + j, channel);
196 a[i] = sum /(float) k;
197 }
198
199 for (i = 0; i < c.num_frames(); ++i)
200 c.a(i,channel) = a[i];
201
202 delete [] a;
203}

251EST_Track differentiate(EST_Track &c, float samp_int)
252{
253 // Differentiate track. SEE ALSO delta(EST_Track, int) which does
254 // this in a more sophisticated way!!!
255
256 EST_Track diff;
257 int i, j;
258 float dist;
259
260 if (samp_int != 0.0)
261 c.sample(samp_int);
262
263 diff.copy_setup(c);
264 diff.resize(c.num_frames() - 1, c.num_channels());
265
266 for (i = 0; i < diff.num_frames(); ++i)
267 {
268 dist = c.t(i + 1) - c.t(i);
269 for (j = 0; j < diff.num_channels(); ++j)
270 diff.a(i, j) = (c.track_break(i) || c.track_break(i + 1)) ? 0.0
271 : (c.a(i + 1) - c.a(i)) / dist;
272 diff.t(i) = c.t(i) + (dist / 2.0);
273 }
274
275 return diff;
276}

372void meansd(EST_Track &tr, float &m, float &sd, int channel)
373{
374 int i, n;
375
376 m = mean( tr, channel );
377
378 float var=0.0;
379 int tr_num_frames = tr.num_frames();
380 for( i=0, n=0; i<tr_num_frames; ++i)
381 if( !tr.track_break(i) ){
382 var += pow(tr.a_no_check(i, channel) - m, float(2.0));
383 ++n;
384 }
385
386 if( n>1 ){ // use n, not tr_num_frames because of breaks
387 var /= (float) (n-1);
388 sd = sqrt(var);
389 }
390 else
391 sd = 0.0;
392}

(b.num_frames()));
430 float predict,real;
431 EST_SuffStats x,y,xx,yy,xy,se,e;
432 float cor,error;
433
434 for (i = 0; i < size; ++i)

439 real = a.a(i, channela);
440 x += predict;
441 y += real;
442 error = predict-real;
443 se += error*error;
444 e += fabs(error);
445 xx += predict*predict;
446 yy += real*real;
447 xy += predict*real;
448 }
449
450 cor = (xy.mean() - (x.mean()*y.mean()))/
451 (sqrt(xx.mean()-(x.mean()*x.mean())) *
452 sqrt(yy.mean()-(y.mean()*y.mean())));
453
454// cout << xy.mean() << " " << x.mean() << " " << y.mean() << " "
455// << xx.mean() << " " << yy.mean() << endl;
456
457 cout << "RMSE " << sqrt(se.mean()) << " Correlation is " << cor
458 << " Mean (abs) Error " << e.mean() << " (" << e.stddev() << ")"
459 << endl;
460 return cor;
461}

601{
602 int i, j, k, l;
603 EST_Track diff;
604 diff = ref;
605 float t;
606
607 // relaxation allows an error to be ignored near boundaries. The
608 // degree of relation specifies how many frames can be ignored.
609
610 float *r = new float[relax*3];
611
612 for (l = 0; l < ref.num_channels(); ++l)
613 for (i = 0; i < ref.num_frames(); ++i)
614 {
615 t = 0;
616 for (k = 0, j = Gof((i - relax), 0)((((i - relax)) > (0)) ? ((i - relax)) : (0)); j < i + relax + 1; ++j, ++k)
617 {
618 if (ref.a(i, l) > 0.5)
619 r[k] = ((j < test.num_frames()) && (test.a(j, l)> 0.6)) ?1
620 : 0.5;
621 else
622 r[k] = ((j < test.num_frames()) && (test.a(j, l)< 0.4)) ? -1
623 : -0.5;
624
625 // fix for relaxation
626 t = r[k];
627 }
628// cout << "ref: " << ref.a(i, l) << " test:" << test.a(i, l) << " error:" << t << endl;
629 diff.a(i, l) = t;
630 }
631
632 delete [] r;
633 return diff;
634}

697int nearest_boundary(EST_Track &tr, float time, int sample_rate, float offset)
698{
699 time -= offset;
700
701 float distance = 10000;
702
703 for (int i = 0; i < tr.num_frames(); ++i)
704 {
705 float start, center, end;
706
707 get_frame(tr, sample_rate, i, start, center, end);
708
709 // printf("nb %f: %d distance %f start %f\n", time, i, distance, start);
710
711 if (fabs(start-time) > distance)
712 return i-1;
713 distance = fabs(start-time);
714 }
715
716 return tr.num_frames();
717}

793int get_order(const EST_Track &t, EST_CoefficientType type, int d)
794{
795 int order;
796 EST_ChannelType start_c = (EST_ChannelType)EST_CoefChannelId(type, d, 0)( (type)<<3 | ((d)<<1 & 6) | ((0)&1) );
797 EST_ChannelType end_c = (EST_ChannelType)EST_CoefChannelId(type, d, 1)( (type)<<3 | ((d)<<1 & 6) | ((1)&1) );
798
799 if (t.has_channel(start_c))
800 if (t.has_channel(end_c))
801 order = t.channel_position(end_c) - t.channel_position(start_c);
802 else
803 order = t.num_channels()-t.channel_position(start_c)-1;
804 else
805 order=0;
806 return order;
807}

815 if ((order=get_order(tr,t))>0)
816 return order;
817
818 cout << "No coefficients in track\n";
819 return 0;
820}

924 num_channels += list(p).num_channels();
925
926 if (style == "first")
927 {
928 num_frames = list.first().num_frames();
929 longest = list.head();
930 }
931 else
932 {
933 if (style != "longest")
934 cerr << "EST_Track: unknown combine style \"" << style <<
935 "\" assuming longest" << endl;
936 for (num_frames = 0, longest = p = list.head(); p; p = p->next())
937 if (num_frames < list(p).num_frames())
938 {
939 num_frames = list(p).num_frames();
940 longest = p;
941 }
942 }
943
944 a.resize(num_frames, num_channels);
945 a.fill(0.0);
