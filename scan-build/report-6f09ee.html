174static EST_read_status load_all_contents(EST_TokenStream &ts,
175// EST_THash<int,EST_Val> &sitems,
176 EST_TVector < EST_Item_Content * > &sitems,
177 int &max_id)
178{
179 // Load items into table with names for later reference
180 // by relations
181 EST_String Sid;
182 bool ok;
183 int id,idval;
184
185 while (ts.peek() != "End_of_Stream_Items")
186 {
187 EST_Item_Content *si = new EST_Item_Content;
188
189 si->relations.add_item("__READ__", est_val((EST_Item *)NULL__null), 1);
190
191 id = 0;
Value stored to 'id' is never read
192
193 Sid = ts.get().string();
194
195 id = Sid.Int(ok);
196 if (!ok)
197 {
198 cerr << "utt_load: " << ts.pos_description() <<
199 " Item name not a number: " << Sid << endl;
200 return misc_read_errorread_error;
201 }
202 if (id >= sitems.length())
203 {
204 sitems.resize(id*2, 1);
205 }
206 sitems[id] = si;
207 // sitems.add_item(id,est_val(si));
208 if (si->f.load(ts) != format_okread_ok)
209 return misc_read_errorread_error;
210 idval = si->f.I("id",0);
211 if (idval > max_id)
212 max_id = idval;
213 if (ts.eof())
214 return misc_read_errorread_error; // just in case this happens
215 }
216
217 ts.get(); // skip "End_of_Stream_Items"
218
219 return format_okread_ok;

237 return misc_read_errorread_error;

252EST_write_status EST_UtteranceFile::save_est_ascii(ostream &outf,const EST_Utterance &utt)
253{
254 EST_write_status v = write_ok;
255
256 outf.precision(8);
257 outf.setf(ios::fixed, ios::floatfield);
258 outf.width(8);
259
260 outf << "EST_File utterance\n"; // EST header identifier.
261 outf << "DataType ascii\n";
262 outf << "version 2\n";
263 outf << "EST_Header_End\n"; // EST end of header identifier.
264
265 // Utterance features
266 outf << "Features ";
267 utt.f.save(outf);
268 outf << endl;
269
270 outf << "Stream_Items\n";
271 EST_TKVL<void *,int> sinames;
272 v = utt_save_all_contents(outf,utt,sinames);
273 if (v == write_fail) return v;
274 outf << "End_of_Stream_Items\n";
275
276 // Relations
277 outf << "Relations\n";
278 EST_Features::Entries p;
279 for (p.begin(utt.relations); p; p++)
280 {
281 v = relation(p->v)->save(outf,sinames);
282 if (v == write_fail) return v;
283 }
284 outf << "End_of_Relations\n";
285
286 outf << "End_of_Utterance\n";
287 return write_ok;
288}

290static EST_write_status utt_save_all_contents(ostream &outf,
291 const EST_Utterance &utt,
292 EST_TKVL<void *,int> &sinames)
293{
294 // Write out all stream items in the utterance, as they may appear in
295 // various places in an utterance keep a record of which ones
296 // have been printed and related them to names for reference by
297 // the Relations (and older Stream architecture).
298 int si_count = 1;
299 EST_write_status v = write_ok;
300
301 // Find the stream items in the relations
302 EST_Features::Entries p;
303 for (p.begin(utt.relations); p; p++)
304 {
305 v = utt_save_all_contents(outf,relation(p->v)->head(),
306 sinames,si_count);
307 if (v == write_fail) return v;
308 }
309
310 return v;
311}

415 EST_Utterance &u,
416 int &max_id)
417{
418 FILE *stream;
419
420 if ((stream=ts.filedescriptor())==NULL__null)
421 return read_error;
422
423 long pos=ftell(stream);
424
425 {
426
427 char buf[81];
428 buf[0] = 0;
429 if (fgets(buf, 80, stream) == NULL__null)
430 {
431 if (ferror(stream)) {
432 cerr << "Error reading xml header" << endl;
433 return read_error;
434 }
435 }
436 if (strncmp(buf, "<?xml", 5) != 0)
437 return read_format_error;
438
439 buf[0] = 0;
440 if (fgets(buf, 80, stream) == NULL__null)
441 {
442 if (ferror(stream)) {
443 cerr << "Error reading DOCTYPE apml header" << endl;
444 return read_error;
445 }
446 }
447
448 if (strncmp(buf, "<!DOCTYPE apml", 14) != 0)
449 return read_format_error;
450 }
451
452 fseek(stream, pos, 0);
453
454 EST_read_status stat = apml_read(stream, ts.filename(),u, max_id);
455
456 if (stat != read_ok)
457 fseek(stream, pos, 0);
458
459 return stat;

501EST_write_status EST_UtteranceFile::save_genxml(ostream &outf,
502 const EST_Utterance &utt)
503{
504 EST_write_status status=write_ok;
505
506 EST_TStringHash<int> features(20);
507
508 EST_Features::Entries p;
509
510 for (p.begin(utt.relations); p; ++p)
511 {
512 EST_Relation *rel = ::relation(p->v);
513
514 EST_Item * hd = rel->head();
515
516 while (hd)
517 {
518 EST_Features::Entries fp;
519 for (fp.begin(hd->features()); fp; ++fp)
520 features.add_item(fp->k, 1);
521 hd=hd->next();
522 }
523 }
524
525 outf << "<?xml version='1.0'?>\n";
526
527 outf << "<!DOCTYPE utterance PUBLIC '//CSTR EST//DTD cstrutt//EN' 'cstrutt.dtd'\n\t[\n";
528
529 EST_TStringHash<int>::Entries f;
530
531 outf << "\t<!ATTLIST item\n";
532 for (f.begin(features); f; ++f)
533 {
534 if (f->k != "id")
535 {
536 outf << "\t\t" << f->k << "\tCDATA #IMPLIED\n";
537 }
538 }
539
540 outf << "\t\t>\n";
541
542 outf << "\t]>\n";
543
544 outf << "<utterance>\n";
545
546 outf << "<language name='unknown'/>\n";
547
548 for (p.begin(utt.relations); p; ++p)
549 {
550 EST_Relation *rel = ::relation(p->v);
551
552 EST_Item * hd = rel->head();
553
554
555 while (hd)
556 {
557 if (hd->up() || hd->down())
558 break;
559 hd=hd->next();
560 }
561
562 // didn't find anything => this is linear
563 if(!hd)
564 {
565 outf << "<relation name='"<< rel->name()<< "' structure-type='list'>\n";
566
567 hd = rel->head();
568 while (hd)

570 outf << " <item\n";
571

576

583 }
