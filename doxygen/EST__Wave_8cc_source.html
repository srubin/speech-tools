198 EST_read_status EST_Wave::load(EST_TokenStream &ts,
199  int offset, int length,
200  int rate)
201 {
202  EST_read_status stat = read_error;
203  int pos = ts.tell();
204 
205  for(int n=0; n< EST_WaveFile::map.n() ; n++)
206  {
207  EST_WaveFileType t = EST_WaveFile::map.token(n);
208 
209  if (t == wff_none)
210  continue;
211 
212  EST_WaveFile::Info *info = &(EST_WaveFile::map.info(t));
213 
214  if (! info->recognise)
215  continue;
216 
217  EST_WaveFile::Load_TokenStream * l_fun =info->load;
218 
219  if (l_fun == NULL)
220  continue;
221 
222  ts.seek(pos);
223  stat = (*l_fun)(ts, *this,
224  rate, st_short, EST_NATIVE_BO, 1,
225  offset, length);
226 
227  if (stat == read_ok)
228  {
229  set_file_type(EST_WaveFile::map.value(t));
230  break;
231  }
232  else if (stat == read_error)
233  break;
234  }
235 
236  return stat;
237 }

503 void EST_Wave::rescale(float gain, int normalize)
504 {
505  int ns;
506  float factor = gain;
507 
508  if (normalize)
509  {
510  int max = 0;
511  for (int i = 0; i < num_samples(); ++i)
512  for (int j = 0; j < num_channels(); ++j)
513  if (abs(a_no_check(i,j)) > max)
514  max = abs(a_no_check(i,j));
515  if (fabs(max/32766.0-gain) < 0.001)
516  return; /* already normalized */
517  else
518  factor *= 32766.0/(float)max;
519  }
520 
521  for (int i = 0; i < num_samples(); ++i)
522  for (int j = 0; j < num_channels(); ++j)
523  {
524  ns = (int)(((float)a_no_check(i,j) * factor) + 0.5);
525  if (ns < -32766)
526  a_no_check(i,j)= -32766;
527  else if (ns > 32766)
528  a_no_check(i,j)= 32766;
529  else
530  a_no_check(i,j)= ns;
531  }
532 }

534 void EST_Wave::rescale( const EST_Track &fc )
535 {
536  int ns, start_sample, end_sample;
537  float target1, target2, increment, factor;
538 
539  int fc_length = fc.length();
540  int _num_channels = num_channels();
541 
542  cerr << ((int)(fc.t(fc_length-1) * p_sample_rate)) << endl;
543 
544  if( ((int)(fc.t(fc_length-1) * p_sample_rate)) > num_samples() )
545  EST_error( "Factor contour track exceeds waveform length (%d samples)",
546  (fc.t(fc_length-1) * p_sample_rate) - num_samples() );
547 
548  start_sample = static_cast<unsigned int>( fc.t( 0 )*p_sample_rate );
549  target1 = fc.a(0,0); // could use separate channels for each waveform channel
550 
551  for ( int k = 1; k<fc_length; ++k ){
552  end_sample = static_cast<unsigned int>( fc.t( k )*p_sample_rate );
553  target2 = fc.a(k);
554 
555  increment = (target2-target1)/(end_sample-start_sample+1);
556 
557  factor = target1;
558  for( int i=start_sample; i<end_sample; ++i, factor+=increment )
559  for( int j=0; j<_num_channels; ++j ){
560  ns = (int)(((float)a_no_check(i,j) * factor) + 0.5);
561  if (ns < -32766)
562  a_no_check(i,j)= -32766;
563  else if (ns > 32766)
564  a_no_check(i,j)= 32766;
565  else
566  a_no_check(i,j)= ns;
567  }
568  start_sample = end_sample;
569  target1 = target2;
570  }
571 }
