268 {
269  // Finds first transition (hopefully deterministic)
270  EST_WFST_State *s = p_states(state);
271  EST_Litem *i;
272 
273  for (i=s->transitions.head(); i != 0; i=i->next())
274  {

276  (out == s->transitions(i)->out_symbol()))
277  {
278  if (cumulate())
279  s->transitions(i)->set_weight(1+s->transitions(i)->weight());
280  return s->transitions(i);

282  }
283 
284  return 0; // no match

287 int EST_WFST::transition(int state,int in, int out,float &prob) const
288 {
289  // Finds first transition (hopefully deterministic)
290  EST_WFST_Transition *trans = find_transition(state,in,out);
291 
292  if (trans == 0)
293  {
294  prob = 0;
295  return WFST_ERROR_STATE;
296  }
297  else
298  {
299  prob = trans->weight();
300  return trans->state();
301  }

304 EST_write_status EST_WFST::save_binary(FILE *fd)
305 {
306  int i;
307  EST_Litem *j;
308  int num_transitions, type, in, out, next_state;
309  float weight;
310 
311  for (i=0; i<p_num_states; i++)
312  {
313  num_transitions = p_states[i]->num_transitions();
314  fwrite(&num_transitions,4,1,fd);
315  if (p_states[i]->type() == wfst_final)
316  type = WFST_FINAL;
317  else if (p_states[i]->type() == wfst_nonfinal)
318  type = WFST_NONFINAL;
319  else if (p_states[i]->type() == wfst_licence)
320  type = WFST_LICENCE;
321  else
322  type = WFST_ERROR;
323  fwrite(&type,4,1,fd);
324  for (j=p_states[i]->transitions.head(); j != 0; j=j->next())
325  {
326  in = p_states[i]->transitions(j)->in_symbol();
327  out = p_states[i]->transitions(j)->out_symbol();
328  next_state = p_states[i]->transitions(j)->state();
329  weight = p_states[i]->transitions(j)->weight();
330 
331  if (in == out)
332  {
333  in *= -1;
334  fwrite(&in,4,1,fd);
335  }
336  else
337  {
338  fwrite(&in,4,1,fd);
339  fwrite(&out,4,1,fd);
340  }
341  fwrite(&next_state,4,1,fd);
342  fwrite(&weight,4,1,fd);
343  }
344  }
345 
346  return write_ok;
347 }

349 EST_write_status EST_WFST::save(const EST_String &filename,
350  const EST_String type)
351 {
352  FILE *ofd;
353  int i;
354  static EST_Regex needquotes(".*[()'\";., \t\n\r].*");
355  EST_Litem *j;
356 
357  if (filename == "-")
358  ofd = stdout;
359  else if ((ofd = fopen(filename,"wb")) == NULL)
360  {
361  cerr << "WFST: cannot write to file \"" << filename << "\"" << endl;
362  return misc_write_error;
363  }
364 
365  fprintf(ofd,"EST_File fst\n");
366  fprintf(ofd,"DataType %s\n",(const char *)type);
367  fprintf(ofd,"in %s\n",
368  (const char *)quote_string(EST_String("(")+
369  p_in_symbols.print_to_string(TRUE)+")",
370  "\"","\\",1));
371  fprintf(ofd,"out %s\n",
372  (const char *)quote_string(EST_String("(")+
373  p_out_symbols.print_to_string(TRUE)+")",
374  "\"","\\",1));
375  fprintf(ofd,"NumStates %d\n",p_num_states);
376  fprintf(ofd, "ByteOrder %s\n", ((EST_NATIVE_BO == bo_big) ? "10" : "01"));
377  fprintf(ofd,"EST_Header_End\n");
378 
379  if (type == "binary")
380  save_binary(ofd);
381  else
382  {
383  for (i=0; i < p_num_states; i++)
384  {
385  EST_WFST_State *s=p_states[i];
386  fprintf(ofd,"((%d ",s->name());
387  switch(s->type())
388  {
389  case wfst_final:
390  fprintf(ofd,"final ");
391  break;
392  case wfst_nonfinal:
393  fprintf(ofd,"nonfinal ");
394  break;
395  case wfst_licence:
396  fprintf(ofd,"licence ");
397  break;
398  default:
399  fprintf(ofd,"error ");
400  }
401  fprintf(ofd,"%d)\n",s->num_transitions());
402  for (j=s->transitions.head(); j != 0; j=j->next())
403  {
404  EST_String in = p_in_symbols.name(s->transitions(j)->in_symbol());
405  EST_String out=p_out_symbols.name(s->transitions(j)->out_symbol());
406  if (in.matches(needquotes))
407  fprintf(ofd," (%s ",(const char *)quote_string(in,"\"","\\",1));
408  else
409  fprintf(ofd," (%s ",(const char *)in);
410  if (out.matches(needquotes))
411  fprintf(ofd," %s ",(const char *)quote_string(out,"\"","\\",1));
412  else
413  fprintf(ofd," %s ",(const char *)out);
414  fprintf(ofd,"%d %g)\n",
415  s->transitions(j)->state(),
416  s->transitions(j)->weight());
417  }
418  fprintf(ofd,")\n");
419  }
420  }
421  if (ofd != stdout)
422  fclose(ofd);
423 
424  return write_ok;
425 }

453 EST_read_status EST_WFST::load_binary(FILE *fd,
454  EST_Option &hinfo,
455  int num_states,
456  int swap)
457 {
458  EST_read_status r;
459  int i,j, s;
460  int num_trans, state_type;
461  int in_sym, out_sym, next_state;
462  float trans_cost;
463 
464  r = format_ok;
465 
466  for (i=0; i < num_states; i++)
467  {
468  num_trans = get_int(fd,swap);
469  state_type = get_int(fd,swap);
470 
471  if (state_type == WFST_FINAL)
472  s = add_state(wfst_final);
473  else if (state_type == WFST_NONFINAL)
474  s = add_state(wfst_nonfinal);
475  else if (state_type == WFST_LICENCE)
476  s = add_state(wfst_licence);
477  else if (state_type == WFST_ERROR)
478  s = add_state(wfst_error);
479  else
480  {
481  cerr << "WFST load: unknown state type \"" <<
482  state_type << "\"" << endl;
483  r = read_format_error;
484  break;
485  }
486 
487  if (s != i)
488  {
489  cerr << "WFST load: internal error: unexpected state misalignment"
490  << endl;
491  r = read_format_error;
492  break;
493  }
494 
495  for (j=0; j < num_trans; j++)
496  {
497  in_sym = get_int(fd,swap);
498  if (in_sym < 0)
499  {
500  in_sym *= -1;
501  out_sym = in_sym;
502  }
503  else
504  out_sym = get_int(fd,swap);
505  next_state = get_int(fd,swap);
506  trans_cost = get_float(fd,swap);
507 
508  p_states[i]->add_transition(trans_cost,next_state,in_sym,out_sym);
509  }
510  }
511 
512  return r;
513 }

516 EST_read_status EST_WFST::load(const EST_String &filename)
517 {
518  // Load a WFST from a file
519  FILE *fd;
520  EST_TokenStream ts;
521  EST_Option hinfo;
522  bool ascii;
523  EST_EstFileType t;
524  EST_read_status r;
525  int i,s;
526  int swap;
527 
528  if ((fd=fopen(filename,"r")) == NULL)
529  {
530  cerr << "WFST load: unable to open \"" << filename
531  << "\" for reading" << endl;
532  return read_error;
533  }
534  ts.open(fd,FALSE);
535  ts.set_quotes('"','\\');
536 
537  if (((r = read_est_header(ts, hinfo, ascii, t)) != format_ok) ||
538  (t != est_file_fst))
539  {
540  cerr << "WFST load: not a WFST file \"" << filename << "\"" <<endl;
541  return misc_read_error;
542  }
543 
544  // Value is a quoted quoted s-expression. Two reads is the
545  // safest way to unquote them
546  LISP inalpha =
547  read_from_string(get_c_string(read_from_string(hinfo.val("in"))));
548  LISP outalpha =
549  read_from_string(get_c_string(read_from_string(hinfo.val("out"))));
550  p_start_state = 0;
551 
552  clear();
553  init(inalpha,outalpha);
554 
555  int num_states = hinfo.ival("NumStates");
556  r = format_ok;
557 
558  if (!ascii)
559  {
560  if (!hinfo.present("ByteOrder"))
561  swap = FALSE; // ascii or not there for some reason
562  else if (((hinfo.val("ByteOrder") == "01") ? bo_little : bo_big)
563  != EST_NATIVE_BO)
564  swap = TRUE;
565  else
566  swap = FALSE;
567  r = load_binary(fd,hinfo,num_states,swap);
568  }
569  else
570  {
571  for (i=0; i < num_states; i++)
572  {
573  LISP sd = lreadf(fd);
574  if (i != get_c_int(car(car(sd))))
575  {
576  cerr << "WFST load: expected description of state " << i <<
577  " but found \"" << siod_sprint(sd) << "\"" << endl;
578  r = read_format_error;
579  break;
580  }
581  if (streq("final",get_c_string(car(cdr(car(sd))))))
582  s = add_state(wfst_final);
583  else if (streq("nonfinal",get_c_string(car(cdr(car(sd))))))
584  s = add_state(wfst_nonfinal);
585  else if (streq("licence",get_c_string(car(cdr(car(sd))))))
586  s = add_state(wfst_licence);
587  else
588  {
589  cerr << "WFST load: unknown state type \"" <<
590  siod_sprint(car(cdr(car(sd)))) << "\"" << endl;
591  r = read_format_error;
592  break;
593  }
594 
595  if (s != i)
596  {
597  cerr << "WFST load: internal error: unexpected state misalignment"
598  << endl;
599  r = read_format_error;
600  break;
601  }
602  if (load_transitions_from_lisp(s,cdr(sd)) != format_ok)
603  {
604  r = read_format_error;
605  break;
606  }
607  }
608  }
609 
610  fclose(fd);
611 
612  return r;
613 }

615 EST_read_status EST_WFST::load_transitions_from_lisp(int s, LISP trans)
616 {
617  LISP t;
618 
619  for (t=trans; t != NIL; t=cdr(t))
620  {
621  float w = get_c_float(siod_nth(3,car(t)));
622  int end = get_c_int(siod_nth(2,car(t)));
623  int in = p_in_symbols.name(get_c_string(siod_nth(0,car(t))));
624  int out = p_out_symbols.name(get_c_string(siod_nth(1,car(t))));
625 
626  if ((in == -1) || (out == -1))
627  {
628  cerr << "WFST load: unknown vocabulary in state transition"
629  << endl;
630  cerr << "WFST load: " << siod_sprint(car(t)) << endl;
631  return read_format_error;
632  }
633  p_states[s]->add_transition(w,end,in,out);
634  }
635  return format_ok;
636 }
