106void srpd(EST_Wave &sig, EST_Track &fz, Srpd_Op &srpd_op, int resize)
107{
108 int i, rns, tracklen, j = 0;
109 SEGMENT_ segment;
110 CROSS_CORR_ cc;
111 STATUS_ pda_status, held_status;
112 srpd_op.sample_freq = sig.sample_rate();
113 float min, max;
114 min = srpd_op.min_pitch; // must store as set up routines corrupt
Value stored to 'min' is never read
115 max = srpd_op.max_pitch;
116
117 initialise_structures (&srpd_op, &segment, &cc);
118 initialise_status (&srpd_op, &pda_status);
119 initialise_status (&srpd_op, &held_status);
120
121 tracklen = (sig.num_samples() - segment.length) / segment.shift + 1;
122
123 if (resize)
124 {
125 fz.set_equal_space(true);
126 fz.resize(tracklen, 1);
127 fz.set_channel_name("F0", 0);
128 fz.fill_time(srpd_op.shift/1000);
129 }
130
131 if (!fz.equal_space())
132 EST_error(EST_error_where = __null), (*EST_error_func)("Pitch tracking algorithm must have equal spaced track\n");
133
134 while ((rns = read_next_wave_segment (sig, &srpd_op, &segment)) != 0)
135 {
136 if (rns == 2)
137 {
138 for (i = 0; i < cc.size; cc.coeff[i++] = 0.0);
139 initialise_status (&srpd_op, &pda_status);
140 }
141 else
142 super_resolution_pda (&srpd_op, segment, &cc, &pda_status);
143 if (pda_status.s_h == HOLD1)
144 {
145 held_status.pitch_freq = pda_status.pitch_freq;
146 held_status.v_uv = VOICED1;
147 held_status.s_h = HELD1;
148 held_status.cc_max = pda_status.cc_max;
149 held_status.threshold = pda_status.threshold;
150 continue;
151 }
152 if (held_status.s_h == HELD1)
153 {
154 if (pda_status.pitch_freq == BREAK_NUMBER0.0)
155 {
156 held_status.pitch_freq = BREAK_NUMBER0.0;
157 held_status.v_uv = UNVOICED0;
158 }
159 held_status.s_h = SENT2;
160 if (held_status.v_uv != VOICED1)
161 fz.set_break(j);
162 fz.a(j++) = held_status.pitch_freq;
163 // printf( "track set: %d (of %d) to %f\n", j-1, fz.length(), held_status.pitch_freq );
164 }
165 if (pda_status.v_uv != VOICED1)
166 fz.set_break(j);
167 fz.a(j++) = pda_status.pitch_freq;
168 //printf( "track set: %d (of %d) to %f\n", j-1, fz.length(), pda_status.pitch_freq );
169 }
170 if (held_status.s_h == HELD1)
171 {
172 held_status.pitch_freq = BREAK_NUMBER0.0;
173 held_status.v_uv = UNVOICED0;
174 fz.set_break(j);
175 fz.a(j++) = held_status.pitch_freq;
176 }
177 end_structure_use (&segment, &cc);
178}

199static void parse_srpd_list(EST_Features &al, struct Srpd_Op *srpd)
200{
201 if (al.present("decimation"))
202 srpd->L = al.I("decimation");
203 if (al.present("min_pitch"))
204 srpd->min_pitch = al.F("min_pitch");
205 if (al.present("max_pitch"))
206 srpd->max_pitch = al.F("max_pitch");
207 if (al.present("pda_frame_shift"))
208 srpd->shift = al.F("pda_frame_shift") * 1000.0;
209 if (al.present("pda_frame_length"))
210 srpd->length = al.F("pda_frame_length") * 1000.0;
211 if (al.present("noise_floor"))
212 srpd->Tsilent = al.I("noise_floor");
213 if (al.present("v2uv_coeff_thresh"))
214 srpd->Thigh = al.F("v2uv_coef_thresh");
215 if (al.present("min_v2uv_coef_thresh"))
216 srpd->Tmin = al.F("min_v2uv_coef_thresh");
217 if (al.present("v2uv_coef_thresh_ratio"))
218 srpd->Tmax_ratio = al.F("v2uv_coef_thresh_ratio");
219 if (al.present("anti_doubling_thresh"))
220 srpd->Tdh = al.F("anti_doubling_thresh");
221 if (al.present("peak_tracking"))
222 srpd->peak_tracking = al.I("peak_tracking");
223 if (al.present("sample_frequency"))
224 srpd->sample_freq = al.I("sample_frequency");
225}
