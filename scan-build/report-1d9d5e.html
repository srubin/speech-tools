75{
76 (void)ishift;
77 (void)startt;
78 int num_frames, num_channels;
79 int swap = FALSE(1==0);
80 int i,j,pos,end;
81 float Start_Time, Record_Freq;
82 EST_Features channels;
83 EST_String c, name, type, size, cname;

86 short sbuff[2];

Value stored to 'num_frames' is never read
89 Start_Time = Record_Freq = 0;

91 if (ts.get() != "SSFF")

95 (ts.get() != "(c)") ||
96 (ts.get() != "SHLRC"))

98 cerr << "ssff load track \"" << ts.filename() << "\": bad header"

100 return misc_read_errorread_error;

102
103 while (ts.peek() != "-----------------")
104 {
105 c = (EST_String)ts.get();
106 if (c == "Comment")
107 ts.get_upto_eoln();
108 else if (c == "Start_Time")
109 {
110 Start_Time = atof(ts.get().string());
111 tr.f_set("Start_Time",Start_Time);
112 }
113 else if (c == "Record_Freq")
114 {
115 Record_Freq = atof(ts.get().string());
116 tr.f_set("Record_Freq",Record_Freq);
117 }
118 else if (c == "Machine")
119 {
120 if (ts.get() == "SPARC")
121 {
122 if (EST_NATIVE_BO((((char *)&est_endian_loc)[0] == 0) ? bo_big : bo_little
) != bo_big)
123 swap = TRUE(1==1);
124 }
125 else if (EST_NATIVE_BO((((char *)&est_endian_loc)[0] == 0) ? bo_big : bo_little
) == bo_big)
126 swap = TRUE(1==1);
127 }
128 else if (c == "Column")
129 {
130 name = (EST_String)ts.get();
131 type = (EST_String)ts.get();
132 size = (EST_String)ts.get();
133 cname = EST_String("Channel_")+itoString(num_channels);
134 channels.set(cname+".name",name);
135 channels.set(cname+".type",type);
136 channels.set(cname+".size",atoi(size));

138 }
139 else if ((c == "window_type") ||
140 (c == "window_duration") ||
141 (c == "lpc_order") ||

145 (c == "frame_duration"))

147 type = (EST_String)ts.get();

150 else if (type == "DOUBLE")

154 }
155 else if (ts.eof())

213 return misc_read_errorread_error;
214 }
215 if (swap)
216 swap_bytes_short(sbuff,1);
217 tr(i,j) = (float)(*sbuff);
218 }
219 else
220 {
221 cerr << "ssff load track \"" << ts.filename() <<
222 "\": unknown channel type value \"" << type << "\"" << endl;
223 return misc_read_errorread_error;
224 }
225 }
226
227 return format_okread_ok;

248EST_write_status EST_TrackFile::save_ssff_ts(FILE *fp, EST_Track tr)
249{
250 int i,j;
251 int need_prob_voice = 0;
252
253 if (tr.equal_space() != 1)
254 {
255 cerr << "ssf save track: can't save variable spaced track as SSFF"
256 << endl;
257 return write_error;
258 }
259
260 fprintf(fp,"SSFF -- (c) SHLRC\n");
261 if (EST_NATIVE_BO((((char *)&est_endian_loc)[0] == 0) ? bo_big : bo_little
) == bo_big)
262 fprintf(fp,"Machine SPARC\n");
263 else
264 fprintf(fp,"Machine IBM-PC\n");
265 if (tr.f_present("Start_Time"))
266 fprintf(fp,"Start_Time %g\n",(double)tr.f_F("Start_Time"));
267 else
268 fprintf(fp,"Start_Time 0.000000\n");
269 // If there are less than two sample points I can't get this value
270 if (tr.f_present("Record_Freq"))
271 fprintf(fp,"Record_Freq %g\n",(double)tr.f_F("Record_Freq"));
272 else if (tr.num_frames() < 2)
273 fprintf(fp,"Record_Freq %d\n", 100);
274 else
275 fprintf(fp,"Record_Freq %g\n", 1/(tr.t(1)-tr.t(0)));
276
277 for (i = 0; i < tr.num_channels(); ++i)
278 fprintf(fp, "Column %s DOUBLE 1\n",(const char *)(tr.channel_name(i)));
279 // For EMULABEL to read this is needs prob_voice too
280 if ((tr.num_channels() == 1) &&
281 (tr.channel_name(0) == "F0"))
282 {
283 need_prob_voice = 1;
284 fprintf(fp, "Column prob_voice DOUBLE 1\n");
285 }
286 EST_Featured::FeatEntries p;
287
288 for (p.begin(tr); p; ++p)
289 {
290 if ((p->k == "Start_Time") ||
291 (p->k == "Record_Freq"))
292 continue;
293 else
294 fprintf(fp, "%s DOUBLE %s\n", (const char *)p->k,
295 (const char *) p->v.String());
296 }
297 fprintf(fp,"-----------------\n");
298 for (i=0; i< tr.num_frames(); i++)
299 {
300 double prob_voice;
301 double dd;
302 for (j=0; j< tr.num_channels(); j++)
303 {
304 dd = tr(i,j);
305 fwrite(&dd,sizeof(double),1,fp);
306 }
307 if (need_prob_voice)
308 {
309 if (tr(i,0) == 0)
310 prob_voice = 0;
311 else
312 prob_voice = 1;
313 fwrite(&prob_voice,sizeof(double),1,fp);
314 }
315 }
316
317 return write_ok;
