45 template<class T>
46 EST_TVector<T> &set(EST_TVector<T> &v, const EST_TList<T> &in)
47 {
48  v.resize(in.length(), FALSE);
49  EST_Litem *p;
50  int i = 0;
51  for (p = in.head(); p!= 0; p = p->next(), ++i)
52  v.a_no_check(i) = in.item(p);
53 
54  return v;
55 }

58 {
59  a.resize(in.length(), FALSE);
60  EST_Litem *p;
61  int i = 0;

63  a[i] = in.item(p);
64 
65  return a;
66 }
