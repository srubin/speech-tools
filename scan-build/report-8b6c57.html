101void set_options(EST_Features &op, EST_Option &al)
102{
103 op.set("frame_shift", DEFAULT_FRAME_SIZE0.001);
104 op.set("frame_length", DEFAULT_FRAME_LENGTH0.008);
105 op.set("preemph", DEFAULT_PREEMPH0.94);
106 op.set("frame_order", DEFAULT_ORDER256);
107
108 if (al.present("-shift"))
109 op.set("frame_shift", al.fval("-shift"));
110
111 if (al.present("-length"))
112 op.set("frame_length", al.fval("-length"));
113
114 if (al.present("-order"))
115 op.set("frame_order", al.fval("-order"));
116
117 if (al.present("-sr"))
118 op.set("sp_range", al.fval("-sr"));
119
120 if (al.present("-w"))
121 op.set("sp_wcut", al.fval("-w"));
122
123 if (al.present("-b"))
124 op.set("sp_bcut", al.fval("-b"));

126 if (al.present("-preemph"))
