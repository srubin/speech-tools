222{
223 EST_TokenStream ts;
224 int i, order;
225 double occur;
226
227 if (ts.open(filename) == -1)
228 return misc_read_errorread_error;
229
230 if (ts.peek().string() != "Ngram_2")
231 {
232 ts.close();
233 return wrong_formatread_format_error;
234 }
235 ts.get(); // skip magic number
236
237 order = atoi(ts.get().string());
238 ts.get_upto_eoln(); // skip to next line
239 EST_StrList vocab;
240 EST_StrList pred_vocab; // may be different
241
242 while (!ts.eoln())
243 vocab.append(ts.get().string());
244 ts.get_upto_eoln(); // skip to next line
245 while (!ts.eoln())
246 pred_vocab.append(ts.get().string());
247
248 if(!n.init(order,EST_Ngrammar::dense,vocab,pred_vocab))
249 {
250 cerr << "Something may be wrong with the vocab lists in '"
251 << filename << "'" << endl;
252 return misc_read_errorread_error;
253 }
254
255 EST_StrVector window(order);
256
257 while(!ts.eof())
258 {
259 for (i=0; i < order; i++)
260 window[i] = ts.get().string();
261 if (ts.get().string() != ":")
262 {
263 cerr << "EST_Ngrammar:load_ngram_cstr_ascii missing colon at filepos "
264 << ts.filepos() << endl;
265 return misc_read_errorread_error;
266 }
267 occur = atof(ts.get().string());
268 n.accumulate(window,occur);
269 if (!ts.eoln())
270 {
271 cerr << "EST_Ngrammar:load_ngram_cstr_ascii expect end of line at filepos "
272 << ts.filepos() << endl;
273 return misc_read_errorread_error;
274 }
275 }
276
277 ts.close();
278
279 return format_okread_ok;

352 num_entries = (freq_data_end-freq_data_start)/sizeof(double);
353 double *dd = new double[num_entries];
354

356 fseek(ifd,freq_data_start,SEEK_SET0);

361 ts.close();

364 }
365 if (swap)
366 swap_bytes_double(dd,num_entries);
367
368 for(j=i=0;i<n.num_states();i++)
369 {
370 if (j >= num_entries)
371 {

373 ts.close();

375 return misc_read_errorread_error;
376 }
377 for (k=n.p_states[i].pdf().item_start();
378 (!n.p_states[i].pdf().item_end(k)) && (j < num_entries) ;
379 k = n.p_states[i].pdf().item_next(k))
380 {
381 n.p_states[i].pdf().set_frequency(k,dd[j]);
382 // Update global info too
383 approx_num_samples += dd[j]; // probably not right
384 n.vocab_pdf.cumulate(k,dd[j]);
385
386 // Number of consecutive occurrences of this frequency as in
387 // dd[j+1] if its a negative number
388 if (j+1 >= num_entries)
389 j++;
390 else if (dd[j+1] < -1)
391 dd[j+1]++;
392 else if (dd[j+1] == -1)
393 j +=2;
394 else
395 j++;
396 }
397 }

400 n.p_num_samples = (int)approx_num_samples;

403

406
407 return format_okread_ok;

553EST_write_status
554save_ngram_htk_ascii(const EST_String filename,
555 EST_Ngrammar &n, double floor)
556{
557
558 ostream *ost;
559
560 // only for bigram
561 if(n.order() != 2)
562 {
563 cerr << "Can only save bigrams in htk_ascii format" << endl;
564 return misc_write_errorwrite_error;
565 }
566
567 if (floor < 0)
568 {
569 cerr << "Negative floor probability does not make sense !" << endl;
570 return misc_write_errorwrite_error;
571 }
572
573 if (filename == "-")
574 ost = &cout;
575 else
576 ost = new ofstream(filename);
577
578 if(!(*ost))
579 return write_fail;
580
581 if(floor * (n.pred_vocab->length()-1) > 1)
582 {
583 floor = 1.0 / (double)(n.pred_vocab->length()-1);
584 cerr << "ERROR : floor is impossibly large, scaling it to ";
585 cerr << floor << endl;
586 }
587
588 int i;
589
590 if(n.p_sentence_start_marker == "")
591 {
592 cerr << "Can't save in HTK format as no sentence start/end tags"
593 << " were given !" << endl;
594 return misc_write_errorwrite_error;
595 }
596
597 // need '!ENTER' (or whatever) as first word- that's HTK for you
598 save_ngram_htk_ascii_sub(n.p_sentence_start_marker,ost,n,floor);
599
600 // the real words
601 for(i=0;i<n.vocab->length();i++)
602 {
603 if ( (n.vocab->name(i) != n.p_sentence_start_marker) &&
604 (n.vocab->name(i) != n.p_sentence_end_marker) &&
605 (n.vocab->name(i) != OOV_MARKER"!OOV") )
606 save_ngram_htk_ascii_sub(n.vocab->name(i),ost,n,floor);
607 }
608
609 if(!n.closed_vocab())
610 save_ngram_htk_ascii_sub(OOV_MARKER"!OOV",ost,n,floor);
611
612 save_ngram_htk_ascii_sub(n.p_sentence_end_marker,ost,n,floor);
613
614 if(ost != &cout)
615 delete ost;
616
617 return write_ok;
618}

736EST_write_status
737save_ngram_cstr_ascii(const EST_String filename, EST_Ngrammar &n,
738 const bool trace, double floor)
739{
740 // awb's format
741 (void)trace;
742 ostream *ost;
743 int i;
744 EST_Litem *k;
745
746 if (filename == "-")
747 ost = &cout;
748 else
749 ost = new ofstream(filename);
750
751 if(!(*ost))
752 return write_fail;
753
754 *ost << "Ngram_2 " << n.order() << endl;
755 for (i=0; i < n.vocab->length(); i++)
756 *ost << n.vocab->name(i) << " ";
757 *ost << endl;
758 for (i=0; i < n.pred_vocab->length(); i++)
759 *ost << n.pred_vocab->name(i) << " ";
760 *ost << endl;
761
762 if (n.representation() == EST_Ngrammar::dense)
763 n.print_freqs(*ost,floor);
764 else if (n.representation() == EST_Ngrammar::backoff)
765 {
766 int total_ngrams = (int)pow(float(n.get_vocab_length()),float(n.order()-1));
767
768 for(i=0;i<total_ngrams;i++)
769 {
770 EST_DiscreteProbDistribution this_pdf;
771 const EST_StrVector this_ngram = n.make_ngram_from_index(i);
772 this_pdf = n.prob_dist(this_ngram);
773
774 for (k=this_pdf.item_start();
775 !this_pdf.item_end(k);
776 k = this_pdf.item_next(k))
777 {
778 double freq;
779 EST_String name;
780 this_pdf.item_freq(k,name,freq);

782 for (int jj=0; jj < this_ngram.n(); jj++)
783 *ost << this_ngram(jj) << " ";

789 if(ost != &cout)
790 delete ost;

833EST_write_status
834save_ngram_cstr_bin(const EST_String filename, EST_Ngrammar &n,
835 const bool trace, double floor)
836{
837
838 if (n.representation() == EST_Ngrammar::sparse)
839 return misc_write_errorwrite_error;
840
841 int i;
842 EST_Litem *k;
843 FILE *ofd;
844 double lfreq = -1;
845 double count = -1;
846 int magic = EST_NGRAMBIN_MAGIC1315402337;
847
848 if (filename == "-")
849 {
850 if ((ofd=stdoutstdout) == NULL__null)
851 return misc_write_errorwrite_error;
852 }
853 else
854 {
855 if ((ofd=fopen(filename,"wb")) == NULL__null)
856 return misc_write_errorwrite_error;
857 }
858
859 fwrite(&magic,sizeof(int),1,ofd);
860 fprintf(ofd,"mBin_2 %d\n",n.order());
861 for (i=0; i < n.vocab->length(); i++)
862 fprintf(ofd,"%s ",(const char *)n.vocab->name(i));
863 fprintf(ofd,"\n");
864 for (i=0; i < n.pred_vocab->length(); i++)
865 fprintf(ofd,"%s ",(const char *)n.pred_vocab->name(i));
866 fprintf(ofd,"\n");
867
868 // We use a simple form of run-length encoding, if consecutive
869 // values are equal only a length is printed. lengths are
870 // negative as frequencies (even smoothed ones) can never be -ve
871
872 if ( trace )
873 cerr << "Saving ..." << endl;
874
875 if (n.representation() == EST_Ngrammar::dense)
876 {
877 for(i=0;i<n.num_states();i++)
878 {
879
880 if ( trace )
881 cerr << "\r" << i*100/n.num_states() << "%";
882
883 for (k=n.p_states[i].pdf().item_start();
884 !n.p_states[i].pdf().item_end(k);
885 k = n.p_states[i].pdf().item_next(k))
886 {
887 double freq;
888 EST_String name;
889 n.p_states[i].pdf().item_freq(k,name,freq);
890 if (freq == 0.0)
891 freq = floor;
892 if (freq == lfreq)
893 count--;
894 else
895 {
896 if (count < -1)
897 fwrite(&count,sizeof(double),1,ofd);
898 fwrite(&freq,sizeof(double),1,ofd);
899 count = -1;
900 }
901 lfreq = freq;
902 }
903 }
904 if (count < -1)
905 fwrite(&count,sizeof(double),1,ofd);
906 }
907 else if (n.representation() == EST_Ngrammar::backoff)
908 {
909 // need to construct pdfs in right order
910 // noting that dense states are indexed s.t. the last
911 // word in the ngram is the least significant 'bit'
912
913 // number of ngrams, excluding last word, is
914 int total_ngrams = (int)pow(float(n.get_vocab_length()),float(n.order()-1));
915
916 for(i=0;i<total_ngrams;i++)
917 {
918
919 if ( trace )
920 cerr << "\r" << i*100/total_ngrams << "%";
921
922 EST_DiscreteProbDistribution this_pdf;
923 const EST_StrVector this_ngram = n.make_ngram_from_index(i);
924 this_pdf = n.prob_dist(this_ngram);
925
926 for (k=this_pdf.item_start();
927 !this_pdf.item_end(k);
928 k = this_pdf.item_next(k))
929 {
930
931 double freq;
932 EST_String name;
933 this_pdf.item_freq(k,name,freq);
934 if (freq == lfreq)
935 count--;
936 else
937 {
938 if (count < -1)
939 fwrite(&count,sizeof(double),1,ofd);
940 fwrite(&freq,sizeof(double),1,ofd);
941 count = -1;
942 }
943 lfreq = freq;
944 }
945
946
947 }
948
949 }
950 if ( trace )
951 cerr << "\r \r" << endl;
952
953 fclose(ofd);
954
955 return write_ok;
956}
