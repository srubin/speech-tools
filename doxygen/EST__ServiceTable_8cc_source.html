110 void EST_ServiceTable::read_table(EST_String socketsFileName)
111 {
112  EST_TokenStream str;
113  FILE *sfile;
114 
115  if ((sfile = fopen(socketsFileName, "r"))==NULL)
116  return;
117 
118  if (str.open(sfile, 1))
119  EST_sys_error("Can't access fringe file '%s'",
120  (const char *)socketsFileName);
121 
122  str.set_SingleCharSymbols("(){}[]#=.");
123 
124  while(!str.eof())
125  {
126  EST_Token tok = str.get();
127 
128  if (tok == "#")
129  {
130  str.get_upto_eoln();
131  continue;
132  }
133 
134  EST_String name = tok;
135 
136  str.must_get(".");
137 
138  tok = str.get();
139  EST_String type = tok;
140 
141  str.must_get("=");
142 
143  EST_Token val = str.get_upto_eoln();
144 
145  if (!entries.t.present(name))
146  {
147  Entry newent;
148  newent.name=name;
149  entries.t.add_item(name, newent);
150  }
151 
152  Entry &entry = entries.t.val(name);
153 
154  if (type=="host")
155  entry.hostname=(EST_String)val;
156  else if (type=="address")
157  entry.address=(EST_String)val;
158  else if (type=="type")
159  entry.type=(EST_String)val;
160  else if (type=="port")
161  entry.port=val;
162  else if (type=="cookie")
163  entry.cookie=(EST_String)val;
164  else
165  EST_warning("Unknown entry field '%s' at %s",
166  (const char *)type,
167  (const char *)str.pos_description());
168  }
169 
170  str.close();
171 }

297 const EST_ServiceTable::Entry &EST_ServiceTable::create(const EST_String name,
298  const EST_String type,
299  int socket)
300 {
301  init_random();
302  Entry entry;
303  long cookie = get_random();
304  struct sockaddr_in sin;
305 
306  socklen_t size=sizeof(struct sockaddr_in);
307 
308  // This only gets the port number
309 
310  if (getsockname(socket, (struct sockaddr *)&sin, &size) != 0)
311  EST_sys_error("Can't find my address");
312 
313 
314  // We have to get an address by looking ourselves up.
315  char myname[100];
316 
317  gethostname(myname, 100);
318 
319  struct hostent *hent = gethostbyname(myname);
320 
321  if (hent == NULL)
322  EST_sys_error("Can't look up my address");
323 
324  // Arbitrarily choose the first address.
325  if (hent->h_addr_list != NULL)
326  memcpy(&(sin.sin_addr.s_addr),hent->h_addr_list[0], sizeof (sin.sin_addr.s_addr));
327 
328  EST_String address= inet_ntoa(sin.sin_addr);
329 
330  entry.name = name;
331  entry.type = type;
332  entry.cookie = EST_String::Number(cookie);
333  entry.port = ntohs(sin.sin_port);
334  entry.address = address;
335  entry.hostname = hent->h_name;
336 
337  entries.t.add_item(name, entry);
338 
339  return entries.t.val(name);
340 }
