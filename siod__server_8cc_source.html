128 static void siod_connect(EST_SiodServer *server)
129 {
130  switch (server->connect())
131  {
132  case connect_ok:
133  break;
134 
135  case connect_not_found_error:
136  EST_sys_error("Can't find host '%s:%d'", (const char *)server->servername(), server->port());
137  break;
138 
139  case connect_not_allowed_error:
140  EST_sys_error("Can't connect to '%s:%d'", (const char *)server->servername(), server->port());
141  break;
142 
143  default:
144  EST_sys_error("Error connecting to '%s:%d'", (const char *)server->servername(), server->port());
145  break;
146  }
147 }

243 LISP siod_remote_command(LISP lserver,
244  LISP sexp)
245 {
246  bool my_server;
247 
248  EST_SiodServer *server = get_server(lserver, my_server);
249 
250  EST_SiodServer::Args args;
251 
252  args.set_val("sexp", est_val(sexp));
253 
254  EST_SiodServer::ResultHandler handler;
255  EST_SiodServer::Result &res = handler.res;
256 
257  if (!server->connected())
258  siod_connect(server);
259 
260  if (!server->execute("scheme", "eval", args, handler))
261  {
262  EST_String err = res.S("ERROR");
263  if (my_server)
264  delete server;
265  return strcons(err.length(), err);
266  }
267 
268  if (my_server)
269  delete server;
270 
271  return scheme(handler.res.val("sexp"));
272 }
